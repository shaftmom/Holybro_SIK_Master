42//-----------------------------------------------------------------------------
43// GenerateDecryptionKey()
44//
45// parameters:
46// encryptionKey - xdata pointer to encryption key data source
47// decryptionKey - xdata pointer to decryption key data destination
48// keySize - enumerated type for key size 0/1/2 = 128/192/256
49//
50// returns:
51// status - 0 for success
52// - 1 for ERROR - Invalid Key size parameter.
53//
54// description:
55//
56// This function will generate a decryption key corresponding to a
57// given encryption key.
58//
59// The encryption key must be stored in xdata prior to calling
60// the GenerateDecryptionKey function.
61//
62// The encryption key must be used for all encryption operations and the
63// decryption key must be used for all decryption operations.
64// (Except CTR Mode which uses the encryption key for both operations.)
65//
66// The AES code examples uses four channels as defined in the DMA_defs.h file.
67// The channel assignments are static and fixed at compile time. They are
68// not assigned dynamically.
69//
70// This function puts the MCU core into idle mode while encryption operation
71// is in process. This function is blocking. The function does not return
72// until after the encrpytion operation has completed.
73//
74// A interrupt is needed to wake up from Idle mode. This requires that the
75// global interrupts are enabled. Also, a small DMA ISR that disables EIE2
76// bit 5 is required in the main module.
77//
78// Note that the extended portion of the extended keys comes out first,
79// before the first 16 bytes. So two DMA operations are used for the
80// AESYOUT channel for an extended key.
81//
82//-----------------------------------------------------------------------------
83GENERATE_DECRYPTION_KEY_STATUS
84 GenerateDecryptionKey (
85 VARIABLE_SEGMENT_POINTER(encryptionKey, U8, SEG_XDATA),
86 VARIABLE_SEGMENT_POINTER(decryptionKey, U8, SEG_XDATA),
87 GENERATE_DECRYPTION_KEY_SIZE keySize)
88{
89 // unions used for compiler independent endianness.
90 UU16 addr; // Union used to access pointer bytes.
91
92 U8 keyLength; // Used to calculate key length in bytes.
93
94 // check first for valid operation
95 if(keySize >= KEY_SIZE_UNDEFINED)
96 {
97 return ERROR_INVALID_PARAMETER;
98 }
99 else
100 {
101 // Calculate key length in bytes based on operation parameter.
102 keyLength = ((keySize + 2) << 3);
103 }
104
105 SFRPAGE = DPPE_PAGE;
106
107 AES0BCFG = 0x00; // disable, for now
108 AES0DCFG = 0x00; // disable for now
109
110 // Disable AES0KIN, AES0BIN, & AES0YOUT channels.
111 DMA0EN &= ~AES0_KBY_MASK;
112
113 // Configure AES key input channel using key pointer.
114 // Set length to calculated key length.
115 // Clear DMA0NMD to disable Key wrapping.
116 // Generating the decryption key only requires encrypting one block.
117 DMA0SEL = AES0KIN_CHANNEL;
118 DMA0NCF = AES0KIN_PERIPHERAL_REQUEST;
119 DMA0NMD = NO_WRAPPING;
120 addr.U16 = (U16)(encryptionKey);
121 DMA0NBAL = addr.U8[LSB];
122 DMA0NBAH = addr.U8[MSB];
123 DMA0NSZH = 0;
124 DMA0NSZL = keyLength;
125 DMA0NAOL = 0;
126 DMA0NAOH = 0;
127
128 // Configure AES Block input channel.
129 // Since the input data does not matter, the base address is set to
130 // 0x0000 and the first 16 bytes of data in xram are used.
131 // Set length to 16.
132 // Clear DMA0NMD to disable Key wrapping.
133 DMA0SEL = AES0BIN_CHANNEL;
134 DMA0NCF = AES0BIN_PERIPHERAL_REQUEST;
135 DMA0NMD = NO_WRAPPING;
136 DMA0NBAL = 0x00;
137 DMA0NBAH = 0x00;
138 DMA0NSZH = 0;
139 DMA0NSZL = 16; // one block
140 DMA0NAOL = 0;
141 DMA0NAOH = 0;
142
143 // Configure AES Y output channel for decryption key.
144 // Enable DMA interrupt
145 // Set length to calculated key length.
146 // Clear DMA0NMD to disable Key wrapping.
147 DMA0SEL = AES0YOUT_CHANNEL;
148 DMA0NCF = AES0YOUT_PERIPHERAL_REQUEST|DMA_INT_EN;
149 DMA0NMD = NO_WRAPPING;
150 addr.U16 = (U16)(decryptionKey);
151 DMA0NBAL = addr.U8[LSB];
152 DMA0NBAH = addr.U8[MSB];
153 DMA0NSZH = 0;
154 DMA0NSZL = keyLength;
155 DMA0NAOH = 0;
156
157 // Configure starting address offset to byte 16 if extended key is used.
158 if (keySize> KEY_SIZE_128_BITS) // if extended key
159 DMA0NAOL = 0x10; // point to extended portion
160 else
161 DMA0NAOL = 0x00; // else point to start
162
163 // Clear KBY (Key, Block, and Y out) bits in DMA0INT sfr using mask.
164 DMA0INT &= ~AES0_KBY_MASK;
165 // Set KBY (Key, Block, and Y out) bits in DMA0EN sfr using mask.
166 // This enables the DMA. But the encyption/decryption operation
167 // won't start until the AES block is enabled.
168 DMA0EN |= AES0_KBY_MASK;
169
170 // Configure AES0DCFG to send inverse key to AES0YOUT
171 AES0DCFG = INVERSE_KEY;
172
173 // Set AES0BCFG according to keySize parameter.
174 AES0BCFG = keySize;
175
176 // Generating the decryption key always uses encryption mode.
177 AES0BCFG |= ENCRYPTION_MODE;
178
179 // Enabled AES module to start encryption operation.
180 AES0BCFG |= AES_ENABLE;
181
182 // enable DMA interrupt to terminate Idle mode
183 EIE2 |= 0x20;
184
185
186 // This do...while loop ensures that the CPU will remain in Idle mode
187 // until AES0YOUT DMA channel transfer is complete.
188 do
189 {
190 #ifdef DMA_TRANSFERS_USE_IDLE
191 PCON |= 0x01; // go to Idle mode
192 #endif
193 } while((DMA0INT & AES0YOUT_MASK)==0);
194
195 if (keySize> KEY_SIZE_128_BITS) // if extended key
196 {
197 // It is reccomended to pause the AES DMA channels
198 // while changing the DMA setup. The AES block cannot
199 // be disabled or the rest of the key would be lost.
200
201 // Pause DMA channels used by AES block.
202 DMA0EN &= ~AES0_KBY_MASK;
203
204 DMA0SEL = AES0YOUT_CHANNEL; // select AES0YOUT DMA ch
205 DMA0NAOL = 0; // reset pointer
206 DMA0NSZL = 0x10; // set length to 16 (128-bits)
207
208 // Clear KBXY (Key, Block, and Y out) bits in DMA0INT sfr using mask.
209 DMA0INT &= ~AES0_KBXY_MASK;
210
211 // Set AES0YOUT DMA enable bit only in DMA0EN sfr using mask.
212 // This enables the DMA and the rest of the key will be written out.
213 DMA0EN |= AES0YOUT_MASK;
214
215 // enable DMA interrupt to terminate Idle mode
216 EIE2 |= 0x20;
217
218 // This do...while loop ensures that the CPU will remain in Idle mode
219 // until AES0YOUT DMA channel transfer is complete.
220 do
221 {
222 #ifdef DMA_TRANSFERS_USE_IDLE
223 PCON |= 0x01; // go to Idle mode
224 #endif
225 } while((DMA0INT & AES0YOUT_MASK)==0);
226 }
227
228 //Clear AES Block
229 AES0BCFG = 0x00;
230 AES0DCFG = 0x00;
231
232 // Clear KBY (Key, Block, and Y out) bits in DMA0EN sfr using mask.
233 DMA0EN &= ~AES0_KBY_MASK;
234 // Clear KBY (Key, Block, and Y out) bits in DMA0INT sfr using mask.
235 DMA0INT &= ~AES0_KBY_MASK;
236
237 return SUCCESS;
238}
