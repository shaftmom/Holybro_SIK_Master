46//-----------------------------------------------------------------------------
47// CTR_EncryptDecrypt()
48//
49// parameters:
50// operation - decryption/encrypt & 128/192/256 options
51// plainText - xdata pointer to plainText
52// cipherText - xdata pointer to cipherText
53// counter - xdata pointer to counter
54// encryptionKey - xdata pointer to encryption key
55//
56// returns:
57// status - 0 for success
58// - 1 for ERROR - Invalid operation parameter.
59//
60// description:
61//
62// This function performs Counter (CTR) Mode Encryption or Decryption of a
63// specified number of 16-byte blocks.
64//
65// The plaintext, and encryption key must be stored in xdata prior to calling
66// an encryption operation.
67//
68// Note that the plainText pointer is a pointer to the source data for an
69// encryption operation and a pointer to the destination data for a decryption
70// operation.
71//
72// The ciphertext and encryption key must be stored in xdata prior to calling
73// an decryption operation. The CTR mode decryption operation uses the AES
74// core in encryption mode. So the encryption key is always used for any CTR
75// operation.
76//
77// The counter should be initialized using a nonce at the start of an
78// encryption sequence.
79//
80// This function will return an error if the operation parameter is invalid.
81// Using the error code is optional, but will aid in debugging.
82//
83// The AES code examples uses four channels as defined in the DMA-defs.h file.
84// The channel assignments are static and fixed at compile time. They are
85// not assigned dynamically.
86//
87// This function puts the MCU core into idle mode while encryption/decryption
88// is in process. This function is blocking. The function does not return
89// until after the encryption/decryption operation has completed.
90//
91// A interrupt is needed to wake up from Idle mode. This requires that the
92// global interrupts are enabled. Also, a small DMA ISR that disables EIE2
93// bit 5 is required in the main module.
94//
95// Note that CTR encryption/decryption requires one DMA transfer per block.
96// It is not possible to encrypt/decrypt multiple blocks using a single DMA
97// transfer because the counter must be incremented between blocks.
98//
99//-----------------------------------------------------------------------------
100CTR_ENCRYPT_DECRYPT_STATUS
101 CTR_EncryptDecrypt (CTR_ENCRYPT_DECRYPT_OPERATION operation,
102 VARIABLE_SEGMENT_POINTER(plainText, U8, SEG_XDATA),
103 VARIABLE_SEGMENT_POINTER(cipherText, U8, SEG_XDATA),
104 VARIABLE_SEGMENT_POINTER(counter, U8, SEG_XDATA),
105 VARIABLE_SEGMENT_POINTER(encryptionKey, U8, SEG_XDATA),
106 U16 blocks)
107{
108 // Unions used for compiler independent endianness.
109 UU16 length; // Length in bytes for all blocks.
110 UU16 addr; // Union used to access pointer bytes.
111
112 U8 keyLength; // Used to calculate key length in bytes.
113
114 // Check first for valid operation.
115 if((operation == DECRYPTION_UNDEFINED)||(operation >= ENCRYPTION_UNDEFINED))
116 {
117 return ERROR_INVALID_PARAMETER;
118 }
119 else
120 {
121 // Calculate key length in bytes based on operation parameter.
122 keyLength = (((operation & 0x03) + 2) << 3);
123 }
124
125 // Calculate plaintext and ciphertext total length.
126 // Using <<4 in lieu of * 16 for code efficiency.
127 length.U16 = (blocks << 4);
128
129 // From this point on, "blocks" is used to count remaining blocks.
130 blocks--;
131
132 SFRPAGE = DPPE_PAGE;
133
134 AES0BCFG = 0x00; // disable, for now
135 AES0DCFG = 0x00; // disable for now
136
137 // Disable AES0KIN, AES0BIN, AES0XIN, & AES0YOUT channels.
138 DMA0EN &= ~AES0_KBXY_MASK;
139
140 // Configure AES key input channel using key pointer.
141 // Set length to calculated key length.
142 // Set DMA0NMD to disable Key wrapping.
143 // This is necessary because we want the Key DMA channel
144 // to stop after the first block.
145
146 addr.U16 = (U16)(encryptionKey);
147 DMA0SEL = AES0KIN_CHANNEL;
148 DMA0NCF = AES0KIN_PERIPHERAL_REQUEST;
149 DMA0NMD = NO_WRAPPING;
150 DMA0NBAL = addr.U8[LSB];
151 DMA0NBAH = addr.U8[MSB];
152 DMA0NSZH = 0;
153 DMA0NSZL = keyLength;
154 DMA0NAOL = 0;
155 DMA0NAOH = 0;
156
157 // AES block input is always the counter value for both CTR encryption
158 // and decryption. Set length to 16 for first block only.
159 // Clear DMA0NMD to disable counter wrapping.
160 // This is necessary because we want the Counter DMA channel
161 // to stop after the first block.
162
163 DMA0SEL = AES0BIN_CHANNEL;
164 DMA0NCF = AES0BIN_PERIPHERAL_REQUEST;
165 DMA0NMD = NO_WRAPPING;
166 addr.U16 = (U16)(counter);
167 DMA0NBAL = addr.U8[LSB];
168 DMA0NBAH = addr.U8[MSB];
169 DMA0NSZL = 16; // one block
170 DMA0NSZH = 0;
171 DMA0NAOL = 0;
172 DMA0NAOH = 0;
173
174 // AES XOR input is plaintext for encryption operation or ciphertext
175 // for decryption operation.
176
177 if(operation & ENCRYPTION_MODE)
178 addr.U16 = (U16)(plainText);
179 else
180 addr.U16 = (U16)(cipherText);
181
182 // Configure AES XOR input channel using corresponding address.
183 // Clear DMA0NMD to disable wrapping.
184
185 DMA0SEL = AES0XIN_CHANNEL;
186 DMA0NCF = AES0XIN_PERIPHERAL_REQUEST;
187 DMA0NMD = NO_WRAPPING;
188 DMA0NBAL = addr.U8[LSB];
189 DMA0NBAH = addr.U8[MSB];
190 DMA0NSZL = 16; // one block
191 DMA0NSZH = 0;
192 DMA0NAOL = 0;
193 DMA0NAOH = 0;
194
195 // AES Y output is ciphertext for encryption operation or
196 // plaintext for decryption operation.
197
198 if(operation & ENCRYPTION_MODE)
199 addr.U16 = (U16)(cipherText);
200 else
201 addr.U16 = (U16)(plainText);
202
203 // Configure AES Y output channel using corresponding address
204 // Set length to 16 for first block only.
205 // Clear DMA0NMD to disable wrapping.
206
207 DMA0SEL = AES0YOUT_CHANNEL;
208 DMA0NCF = AES0YOUT_PERIPHERAL_REQUEST|DMA_INT_EN;
209 DMA0NMD = NO_WRAPPING;
210
211 DMA0NBAL = addr.U8[LSB];
212 DMA0NBAH = addr.U8[MSB];
213 DMA0NSZL = 16; // one block
214 DMA0NSZH = 0;
215 DMA0NAOH = 0;
216 DMA0NAOL = 0;
217
218 // Clear KBXY (Key, Block, X in, and Y out) bits in DMA0INT sfr using mask.
219 DMA0INT &= ~AES0_KBXY_MASK;
220
221
222 // Set KBXY (Key, Block, Xin, and Y out) bits in DMA0EN sfr using mask.
223 // This enables the DMA. But the encryption/decryption operation
224 // won't start until the AES block is enabled.
225 DMA0EN |= AES0_KBXY_MASK;
226
227 // Configure data path for XOR on output. CTR mode always has the
228 // XOR on the output for both encryption and decryption.
229 AES0DCFG = XOR_ON_OUTPUT;
230
231 // Configure AES0BCFG for encryption/decryption and requested key size.
232 // Note the encryption mode bit is set explicitly.
233 AES0BCFG = operation;
234 AES0BCFG |= ENCRYPTION_MODE;
235 AES0BCFG |= AES_ENABLE;
236
237 // Enable DMA interrupt to terminate Idle mode.
238 EIE2 |= 0x20;
239
240 // This do...while loop ensures that the CPU will remain in Idle mode
241 // until AES0YOUT DMA channel transfer is complete.
242 do
243 {
244 #ifdef DMA_TRANSFERS_USE_IDLE
245 PCON |= 0x01; // go to Idle mode
246 #endif
247 } while((DMA0INT&AES0YOUT_MASK)==0);
248
249 while(blocks--) // if blocks remaining
250 {
251 // It is necessary to either pause the AES DMA channels
252 // or disable the AES block while changing the setup.
253 // Both steps are taken in this example to be extra safe.
254
255 // Disable AES block.
256 // This also clears the AES contents and resets the state machine.
257 // It is not necessary to reset the AES core between blocks.
258 // But it is recommended to disable the core while changing the set-up.
259
260 AES0BCFG &= ~AES_ENABLE;
261
262 // Pause DMA channels used by AES block.
263 DMA0EN &= ~AES0_KBXY_MASK;
264
265 IncrementCounter(counter);
266
267 SFRPAGE = DPPE_PAGE;
268
269 // AESKIN DMA channel reset address offset
270 DMA0SEL = AES0KIN_CHANNEL;
271 DMA0NAOL = 0;
272 DMA0NAOH = 0;
273
274 // AESBIN DMA channel reset address offset
275 DMA0SEL = AES0BIN_CHANNEL;
276 DMA0NAOL = 0;
277 DMA0NAOH = 0;
278
279 // Ch 2 AES0XIN increment size by 16 bytes
280 DMA0SEL = AES0XIN_CHANNEL;
281 length.U8[LSB] = DMA0NSZL;
282 length.U8[MSB] = DMA0NSZH;
283 length.U16 += 16;
284 DMA0NSZL = length.U8[LSB];
285 DMA0NSZH = length.U8[MSB];
286
287 // Ch 3 AESYOUT increment size by 16 bytes
288 DMA0SEL = AES0YOUT_CHANNEL;
289 length.U8[LSB] = DMA0NSZL;
290 length.U8[MSB] = DMA0NSZH;
291 length.U16 += 16;
292 DMA0NSZL = length.U8[LSB];
293 DMA0NSZH = length.U8[MSB];
294
295 // Clear KBXY (Key, Block, X in, and Y out) bits in DMA0INT sfr using mask.
296 DMA0INT &= ~AES0_KBXY_MASK;
297
298 // Set KBXY (Key, Block, Xin, and Y out) bits in DMA0EN sfr using mask.
299 // This enables the DMA. But the encryption/decryption operation
300 // won't start until the AES block is enabled.
301 DMA0EN |= AES0_KBXY_MASK;
302
303 // Enabled AES module to start encryption/decryption operation.
304 AES0BCFG |= AES_ENABLE;
305
306 // enable DMA interrupt to terminate Idle mode
307 EIE2 |= 0x20;
308
309 // This do...while loop ensures that the CPU will remain in Idle mode
310 // until AES0YOUT DMA channel transfer is complete.
311 do
312 {
313 #ifdef DMA_TRANSFERS_USE_IDLE
314 PCON |= 0x01; // go to Idle mode
315 #endif
316 } while((DMA0INT & AES0YOUT_MASK)==0);
317 }
318
319 //Clear AES Block
320 AES0BCFG = 0x00;
321 AES0DCFG = 0x00;
322
323 // Clear KXBY (Key, Block, XOR, and Y out) bits in DMA0EN sfr using mask.
324 DMA0EN &= ~AES0_KBXY_MASK;
325
326 // Clear KBY (Key, Block, and Y out) bits in DMA0INT sfr using mask.
327 DMA0INT &= ~AES0_KBXY_MASK;
328
329 return SUCCESS;
330}
331//-----------------------------------------------------------------------------
332// IncrementCounter()
333//
334// This function is used to increment the 16-byte counter.
335//
336// This version is hand optimized to produce efficient assembler.
337// This results in obtuse C code.
338//
339//-----------------------------------------------------------------------------
340void IncrementCounter (VARIABLE_SEGMENT_POINTER(counter, U8, SEG_XDATA))
341{
342 U8 i;
343 U8 x;
344
345 i = 16;
346 counter += 16; // point to end of data
347
348 do
349 {
350 counter--; // decrement data pointer
351 x = *counter; // read xdata using data pointer
352 x++; // increment value
353 *counter = x; // move to xram
354 if(x) break; // break if not zero
355 } while(--i); // DJNZ
356}
