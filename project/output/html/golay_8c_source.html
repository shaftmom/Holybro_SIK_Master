41// intermediate arrays for encodeing/decoding. Using these
42// saves some interal memory that would otherwise be needed
43// for pointers
44static __pdata uint8_t g3[3], g6[6];
45
46// encode 3 bytes data into 6 bytes of coded data
47// input is in g3[], output in g6[]
48static void
49golay_encode24(void)
50{
51 __pdata uint16_t v;
52 __pdata uint16_t syn;
53
54 v = g3[0] | ((uint16_t)g3[1] & 0x0F) << 8;
55 syn = golay23_encode[v];
56 g6[0] = syn & 0xFF;
57 g6[1] = (g3[0] & 0x1F) << 3 | syn >> 8;
58 g6[2] = (g3[0] & 0xE0) >> 5 | (g3[1] & 0x0F) << 3;
59
60 v = g3[2] | ((uint16_t)g3[1] & 0xF0) << 4;
61 syn = golay23_encode[v];
62 g6[3] = syn & 0xFF;
63 g6[4] = (g3[2] & 0x1F) << 3 | syn >> 8;
64 g6[5] = (g3[2] & 0xE0) >> 5 | (g3[1] & 0xF0) >> 1;
65}

67// encode n bytes of data into 2n coded bytes. n must be a multiple 3
68// encoding takes about 6 microseconds per input byte
69void
70golay_encode(__pdata uint8_t n, __xdata uint8_t * __pdata in, __xdata uint8_t * __pdata out)
71{
72 while (n >= 3) {
73 g3[0] = in[0]; g3[1] = in[1]; g3[2] = in[2];
74 golay_encode24();
75 out[0] = g6[0]; out[1] = g6[1]; out[2] = g6[2];
76 out[3] = g6[3]; out[4] = g6[4]; out[5] = g6[5];
77 in += 3;
78 out += 6;
79 n -= 3;
80 }
81}

83// decode 6 bytes of coded data into 3 bytes of original data
84// input is in g6[], output in g3[]
85// returns the number of words corrected (0, 1 or 2)
86static uint8_t
87golay_decode24(void)
88{
89 __data uint16_t v;
90 __data uint16_t syn;
91 __data uint16_t e;
92 __pdata uint8_t errcount = 0;
93
94 v = (g6[2] & 0x7F) << 5 | (g6[1] & 0xF8) >> 3;
95 syn = golay23_encode[v];
96 syn ^= g6[0] | (g6[1] & 0x07) << 8;
97 e = golay23_decode[syn];
98 if (e) {
99 errcount++;
100 v ^= e;
101 }
102 g3[0] = v & 0xFF;
103 g3[1] = v >> 8;
104
105 v = (g6[5] & 0x7F) << 5 | (g6[4] & 0xF8) >> 3;
106 syn = golay23_encode[v];
107 syn ^= g6[3] | (g6[4] & 0x07) << 8;
108 e = golay23_decode[syn];
109 if (e) {
110 errcount++;
111 v ^= e;
112 }
113 g3[1] |= (v >> 4) & 0xF0;
114 g3[2] = v & 0xFF;
115
116 return errcount;
117}

119// decode n bytes of coded data into n/2 bytes of original data
120// n must be a multiple of 6
121// decoding takes about 4 microseconds per input byte
122// the number of 12 bit words that required correction is returned
123uint8_t
124golay_decode(__pdata uint8_t n, __xdata uint8_t * __pdata in, __xdata uint8_t * __pdata out)
125{
126 __pdata uint8_t errcount = 0;
127 while (n >= 6) {
128 g6[0] = in[0]; g6[1] = in[1]; g6[2] = in[2];
129 g6[3] = in[3]; g6[4] = in[4]; g6[5] = in[5];
130 errcount += golay_decode24();
131 out[0] = g3[0]; out[1] = g3[1]; out[2] = g3[2];
132 in += 6;
133 out += 3;
134 n -= 6;
135 }
136 return errcount;
137}
