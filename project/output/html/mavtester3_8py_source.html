194def send_override():

226def recv_vehicle(vehicle):
227 '''
228 receive packets in the vehicle
229 '''
230 try:
231 m = vehicle.queue.get(block=False)
232 except Queue.Empty:
233 return False
234 if m.get_type() == 'BAD_DATA':
235 stats.vehicle_bad_data += 1
236 return True
237 if opts.show:
238 print(m)
239 vehicle.received += 1
240 if m.get_type() in ['RADIO','RADIO_STATUS']:
241 #print('VRADIO: ', str(m))
242 vehicle.radio_received += 1
243 stats.vehicle_txbuf = m.txbuf
244 stats.vehicle_fixed = m.fixed
245 if m.get_type() == 'RC_CHANNELS_OVERRIDE':
246 process_override(m)
247 return True

249def recv_GCS():
250 '''
251 receive packets in the GCS
252 '''
253 try:
254 m = gcs_queue.get(block=False)
255 except Queue.Empty:
256 return False
257 if m.get_type() == 'BAD_DATA':
258 stats.gcs_bad_data += 1
259 return True
260 if m.get_type() == 'GLOBAL_POSITION_INT':
261 global gcs_lat
262 src_system = m.get_srcSystem()
263 if gcs_lat[src_system] != m.lat:
264 print("Lost %u GLOBAL_POSITION_INT messages" % (m.lat - gcs_lat[src_system]))
265 gcs_lat[src_system] = m.lat
266 gcs_lat[src_system] += 1
267 if opts.show:
268 print(m)
269 stats.gcs_received += 1
270 if m.get_type() in ['RADIO','RADIO_STATUS']:
271 #print('GRADIO: ', str(m))
272 stats.gcs_radio_received += 1
273 stats.gcs_txbuf = m.txbuf
274 stats.gcs_fixed = m.fixed
275 return True
