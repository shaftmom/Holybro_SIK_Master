172void
173rtc_write_count(uint32_t count)
174{
175 union {
176 uint8_t b[4];
177 uint32_t l;
178 } mix;
179
180 // write the counter shadow
181 mix.l = count;
182 rtc_write_reg(RTC_CAPTURE0, mix.b[0]);
183 rtc_write_reg(RTC_CAPTURE1, mix.b[1]);
184 rtc_write_reg(RTC_CAPTURE2, mix.b[2]);
185 rtc_write_reg(RTC_CAPTURE3, mix.b[3]);
186
187 // load the new value into the counter
188 rtc_write_reg(RTC_CN, CN_LOAD);
189}

191static uint8_t
192rtc_read_reg(uint8_t reg)
193{
194 while (RTC0ADR & RTC0ADR_BUSY) ; // wait for busy to clear
195 RTC0ADR = RTC0ADR_BUSY | reg;
196 while (RTC0ADR & RTC0ADR_BUSY) ; // wait for busy to clear
197 return RTC0DAT;
