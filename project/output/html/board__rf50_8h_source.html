68// board-specific hardware config
69#define HW_INIT \
70 do { \
71 /* GPIO config */ \
72 P0SKIP |= 0x40; /* button */ \
73 P1SKIP |= 0x60; /* buttons */ \
74 P2SKIP |= 0x21; /* LEDs */ \
75 SFRPAGE = CONFIG_PAGE; \
76 P2DRV |= 0x21; /* LEDs */ \
77 SFRPAGE = LEGACY_PAGE; \
78 /* INT0 is the radio interrupt, on P0.7 */ \
79 IT01CF = (IT01CF & 0xf) | 0x7; \
80 IT0 = 0; /* INT0 leve triggered */ \
81 } while(0)
