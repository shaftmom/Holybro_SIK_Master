97// board-specific hardware config
98#define HW_INIT \
99 do { \
100 /* GPIO config */ \
101 P0SKIP |= 0xCF; /* P0 UART avail on XBAR */ \
102 P1SKIP |= 0x78; /* P1 SPI1, CEX0 avail on XBAR */ \
103 P2SKIP |= 0xFF; /* P2 nothing avail on XBAR, All GPIO */ \
104 SFRPAGE = CONFIG_PAGE; \
105 P1MDOUT |= 0xF5; /* SCK1, MOSI1, MISO1 push-pull was 60 */ \
106 P1DRV |= 0xF5; /* SPI signals use high-current mode, LEDs and PAEN High current drive was 60 */ \
107 P2MDOUT |= 0x20; /* PA_ENABLE (P2.5) output */ \
108 P2DRV |= 0x20; /* PA_ENABLE (P2.5) high current drive*/ \
109 SFRPAGE = LEGACY_PAGE; \
110 /* INT0 is the radio interrupt, on P0.7 */ \
111 IT01CF = (IT01CF & 0xf) | 0x7; \
112 IT0 = 0; /* INT0 level triggered */ \
113 } while(0)
114

119 P0DRV |= 0x04; /* CTS */ \
120 SFRPAGE = LEGACY_PAGE; \
121 } while(0)
