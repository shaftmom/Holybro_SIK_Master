74 # Send raw text to radio
75 def __send(self, text):
76 if (self.port is None) or (self.ser is None):
77 return False
78
79 try:
80 res = self.ser.send(text)
81 time.sleep(0.2) # Let the serial line catch up
82 return res
83 except:
84 return False

86 # Form an AT command and send to radio
87 def __send_at(self, command):
88 # Don't send bytes if in "normal" mode, other radio is listening!
89 if not self.is_command:
90 return False
91
92 prefix = ATCommandSet.AT_LOCAL_PREFIX
93 if self.is_remote and (command != ATCommandSet.AT_EXIT):
94 prefix = ATCommandSet.AT_REMOTE_PREFIX
95 text = '\r\n' + prefix + str(command) + '\r\n'
96 return not not self.__send(text)
97
98 # Look for 'pattern' (string RE) and return MatchObject if seen before read_timeout
99 def __expect(self, pattern_list):
100 if (self.port is None) or (self.ser is None):
101 return False
102
103 try:
104 self.ser.expect(pattern_list, timeout=self.read_timeout)
105 res = self.ser.match
106 time.sleep(0.2) # Let the serial line catch up
107 return res
108 except:
109 return False
110
111 # Send AT command, then look for pattern
112 def __query(self, command, pattern):
113 if not self.__send_at(command):
114 return False
115 val = self.__expect(pattern)
116 return val
117
118 # Query for an int
119 def __query_int(self, command):
120 val = self.__query(command, ['(\d+)\r\n'])
121 if val:
122 return int(val.group(0))
123 return False
124
125 # Query for a float
126 def __query_float(self, command):
127 val = self.__query(command, ['(\d+\.\d+)\r\n'])
128 if val:
129 return float(val.group(0))
130 return False
131
132 # Query for literal text (return True if found)
133 def __query_exact(self, command, text):
134 return not not self.__query(command, ['%s\r\n' % text])

139 def enter_command_mode(self):
140 # Technically okay to resend this command, but won't see an 'OK' back
141 if self.is_command:
142 return False
143
144 # Will raise a timeout exception if already in command mode
145 # (due to another process leaving it that way?)
146 time.sleep(1)
147 if not self.__send('+++'):
148 return False
149 time.sleep(1)
150 if not self.__expect(['OK']):
151 return False
152
153 self.is_command = True
154 return True
