154bool
155radio_receive_packet(uint8_t *length, __xdata uint8_t * __pdata buf)
156{
157#ifdef INCLUDE_GOLAY
158 __xdata uint8_t gout[3];
159 __data uint16_t crc1, crc2;
160 __data uint8_t errcount = 0;
161 __data uint8_t elen;
162#endif
163
164 if (!packet_received)
165 return false;
166
167 if (receive_packet_length > MAX_RAW_PACKET_LENGTH
168 || receive_packet_length < 3) {
169 EX0=0;
170 _radio_receiver_on();
171 EX0=1;
172 goto failed;
173 }
174
175
176#ifdef INCLUDE_GOLAY
177 if (!feature_golay)
178#endif // INCLUDE_GOLAY
179 {
180 *length = receive_packet_length-3;
181 memcpy(buf, radio_buffer+3, receive_packet_length-3);
182
183 EX0=0;
184 _radio_receiver_on();
185 EX0=1;
186
187
188 //debug("packet received len=%d chan=%d", *length, settings.current_channel);
189
190 return true;
191 }
192
193#ifdef INCLUDE_GOLAY
194 // decode it in the callers buffer. This relies on the
195 // in-place decode properties of the golay code. Decoding in
196 // this way allows us to overlap decoding with the next receive
197 memcpy(buf, radio_buffer+1, receive_packet_length-1);
198
199 // enable the receiver for the next packet. This also
200 // enables the EX0 interrupt
201 elen = receive_packet_length-1;
202 EX0=0;
203 _radio_receiver_on();
204 EX0=1;
205
206 if (elen < 12 || (elen%6) != 0) {
207 // not a valid length
208 debug("rx len invalid %u\n", (unsigned)elen);
209 goto failed;
210 }
211
212 // decode the header
213 errcount = golay_decode(6, buf, gout);
214 if (gout[0] != netid[0] ||
215 gout[1] != netid[1]) {
216 // its not for our network ID
217 debug("netid %x %x\n",
218 (unsigned)gout[0],
219 (unsigned)gout[1]);
220 goto failed;
221 }
222
223 if (6*((gout[2]+2)/3+2) != elen) {
224 debug("rx len mismatch1 %u %u\n",
225 (unsigned)gout[2],
226 (unsigned)elen);
227 goto failed;
228 }
229
230 // decode the CRC
231 errcount += golay_decode(6, &buf[6], gout);
232 crc1 = gout[0] | (((uint16_t)gout[1])<<8);
233
234 if (elen != 12) {
235 errcount += golay_decode(elen-12, &buf[12], buf);
236 }
237
238 *length = gout[2];
239
240 crc2 = crc16(*length, buf);
241
242 if (crc1 != crc2) {
243 debug("crc1=%x crc2=%x len=%u [%x %x]\n",
244 (unsigned)crc1,
245 (unsigned)crc2,
246 (unsigned)*length,
247 (unsigned)buf[0],
248 (unsigned)buf[1]);
249 goto failed;
250 }
251
252 if (errcount != 0) {
253 if ((uint16_t)(0xFFFF - errcount) > errors.corrected_errors) {
254 errors.corrected_errors += errcount;
255 } else {
256 errors.corrected_errors = 0xFFFF;
257 }
258 if (errors.corrected_packets != 0xFFFF) {
259 errors.corrected_packets++;
260 }
261 }
262
263 return true;
264#endif // INCLUDE_GOLAY
265
266failed:
267 inc_rx_error();
268 return false;
269}

311static bool
312in_rx_mode(void) __reentrant
313{
314 uint8_t state;
315 EX0_SAVE_DISABLE;
316 frr_d_read(state);
317 if (state == STATE_RX) {
318 EX0_RESTORE;
319 return true;
320 }
321 EX0_RESTORE;
322 return false;

325static bool
326radio_transmit_simple(uint8_t length, __xdata uint8_t * __pdata buf, __pdata uint16_t timeout_ticks, bool insert_netid)
327{
328 __pdata uint16_t tstart;
329 __data uint8_t n, len_remaining;
330 uint8_t chip_status, ph_status, tx_space;
331
332 tstart = timer2_tick();
333
334 EX0_SAVE_DISABLE;
335 /* exit from RX state */
336 cmd_change_state(STATE_READY);
337 wait_for_cts();
338
339 EX0_RESTORE;
340
341 cmd_get_int_status_clear_all();
342 wait_for_cts();
343
344 cmd_fifo_info(0x03); /* clear the TX FIFO */
345 wait_for_cts();
346
347 if (insert_netid) {
348 radio_buffer[0] = netid[1];
349 radio_buffer[1] = netid[0];
350 radio_buffer[2] = length;
351 write_tx_fifo(3, radio_buffer);
352 } else {
353 //__xdata uint8_t length_xdata;
354 //radio_buffer[0] = length;
355 write_tx_fifo(1, &length);
356 }
357
358 n = length;
359 if (n > 32)
360 n = 32;
361 write_tx_fifo(n, buf);
362 len_remaining = length - n;
363 buf += n;
364
365 if (insert_netid) {
366 cmd_start_tx(settings.current_channel, 0x30, length+3, 0, 0);
367 } else {
368 cmd_start_tx(settings.current_channel, 0x30, length+1, 0, 0);
369 }
370 wait_for_cts();
371
372 while ((uint16_t) (timer2_tick() - tstart) < timeout_ticks) {
374 cmd_fifo_info(0x00);
375 fifo_info_reply(_skip, tx_space);
376 frr_bc_read(chip_status, ph_status);
377
378 // looking at PH_STATUS_TX_FIFO_ALMOST_EMPTY proved unreliable
379 if (//(ph_status & PH_STATUS_TX_FIFO_ALMOST_EMPTY)
380 (tx_space >= TX_FIFO_THRESHOLD)
381 && len_remaining != 0) {
382 n = len_remaining;
383 if (n > TX_FIFO_THRESHOLD)
384 n = TX_FIFO_THRESHOLD;
385 write_tx_fifo(n, buf);
386 len_remaining -= n; buf += n;
387 }
388
389 if (chip_status & (CHIP_STATUS_FIFO_UNDERFLOW_OVERFLOW_ERROR \
390 | CHIP_STATUS_CMD_ERROR)) {
391 debug("tx error: chip_status=%x len=%u len_remaining=%u", chip_status, length, len_remaining);
392 goto error;
393 }
394
395 if (ph_status & PH_STATUS_PACKET_SENT) {
396 return true;
397 }
398 }
399
400 debug("TX timeout %u ts=%u tn=%u len=%u",
401 timeout_ticks,
402 tstart,
403 timer2_tick(),
404 (unsigned)length);
405
406error:
407 /* exit from TX state */
408 cmd_change_state(STATE_READY);
409 wait_for_cts();
410
411 /* here we assume we are out of TX state, so that the FIFO reset is safe to do */
412 cmd_fifo_info(0x03);
413 wait_for_cts();
414
415 inc_tx_error();
416 return false;
417}

419#ifdef INCLUDE_GOLAY
420// start transmitting a packet from the transmit FIFO
421//
422// @param length number of data bytes to send
423// @param timeout_ticks number of 16usec RTC ticks to allow
424// for the send
425//
426// @return true if packet sent successfully
427//
428static bool
429radio_transmit_golay(uint8_t length, __xdata uint8_t * __pdata buf, __pdata uint16_t timeout_ticks)
430{
431 __pdata uint16_t crc;
432 __xdata uint8_t gin[3];
433 __pdata uint8_t elen, rlen;
434
435 if (length > (sizeof(radio_buffer)/2)-6) {
436 debug("golay packet size %u\n", (unsigned)length);
437 panic("oversized golay packet");
438 }
439
440 // rounded length
441 rlen = ((length+2)/3)*3;
442
443 // encoded length
444 elen = (rlen+6)*2;
445
446 // start of packet is network ID and packet length
447 gin[0] = netid[0];
448 gin[1] = netid[1];
449 gin[2] = length;
450
451 // golay encode the header
452 golay_encode(3, gin, radio_buffer);
453
454 // next add a CRC, we round to 3 bytes for simplicity, adding
455 // another copy of the length in the spare byte
456 crc = crc16(length, buf);
457 gin[0] = crc&0xFF;
458 gin[1] = crc>>8;
459 gin[2] = length;
460
461 // golay encode the CRC
462 golay_encode(3, gin, &radio_buffer[6]);
463
464 // encode the rest of the payload
465 golay_encode(rlen, buf, &radio_buffer[12]);
466
467 return radio_transmit_simple(elen, radio_buffer, timeout_ticks, false);
468}
469#endif // INCLUDE_GOLAY
470
471// start transmitting a packet from the transmit FIFO
472//
473// @param length number of data bytes to send
474// @param timeout_ticks number of 16usec RTC ticks to allow
475// for the send
476//
477// @return true if packet sent successfully
478//
479bool
480radio_transmit(uint8_t length, __xdata uint8_t * __pdata buf, __pdata uint16_t timeout_ticks)
481{
482 bool ret;
483
484 EX0_SAVE_DISABLE;
485
486#ifdef INCLUDE_GOLAY
487 if (!feature_golay) {
488 ret = radio_transmit_simple(length, buf, timeout_ticks, true);
489 } else {
490 ret = radio_transmit_golay(length, buf, timeout_ticks);
491 }
492#else
493 ret = radio_transmit_simple(length, buf, timeout_ticks, true);
494#endif // INCLUDE_GOLAY
495
496 EX0_RESTORE;
497 return ret;
498}

500static void
501_radio_receiver_on(void) __reentrant
502{
503 packet_received = 0;
504 receive_packet_length = 0;
505 partial_packet_length = 0;
506
507 cmd_get_int_status_clear_all();
508 wait_for_cts();
509
510 clear_rx_fifo();
511
512 cmd_start_rx(
513 settings.current_channel,
514 0,
515 0,
516 0, /* no timeout when looking for preamble */
517 STATE_READY, /* go to READY state after reception of valid packet */
518 STATE_READY /* ditto after invalid packet */
519 );
520 wait_for_cts();
521 sync_detected = 0;
522}

554static void radio_gpio_init()
555{
556 cmd_set_property4(GROUP_FRR_CTL, 0x0,
557 10, /* FRR A = latched RSSI */
558 7, /* FRR B = INT_CHIP_STATUS */
559 3, /* FRR C = INT_PH_STATUS */
560 9 /* FRR D = CURRENT_STATE */
561 );
562 wait_for_cts();
563
564 cmd_set_property4(GROUP_INT_CTL, 0x0,
565 0x03, /* MODEM_INT_STATUS_EN | PH_INT_STATUS_EN */
566 RX_INT0_PH_MASK,
567 RX_INT0_MODEM_MASK,
568 0x00
569 );
570 wait_for_cts();
571
572 /* GLOBAL_XO_TUNE */
573 cmd_set_property1(GROUP_GLOBAL, 0x00, EZRADIOPRO_OSC_CAP_VALUE);
574 wait_for_cts();
575
576 /* GLOBAL_CONFIG: fast switch to TX, shared FIFO mode */
577 cmd_set_property1(GROUP_GLOBAL, 0x03, 0x10);
578 wait_for_cts();
579
580 cmd_set_property1(GROUP_GLOBAL, 0x01, 0x48);
581 wait_for_cts();
582
583 cmd_gpio_pin_cfg(
584 0x40 | GPIO_0_CONFIG,
585 0x40 | GPIO_1_CONFIG,
586 0x40 | 8, // 2: CTS
587 0x40 | 33, // GPIO4: RX state
588 39, // nIRQ
589 0, // SDO
590 2
591 );
592 wait_for_cts();
593}

631static void
632rx_hop(uint8_t channel)
633{
634 uint16_t vco_cnt;
635
636 EX0_SAVE_DISABLE;
637 clear_rx_fifo();
638 uint32_t fc = (freq_control_base + channel*((uint32_t) freq_control_spacing));
639 /* formula for VCO_CNT taken from Silabs' spreadsheet calculator */
640 vco_cnt = (uint16_t) (((fc<<5) - (((uint32_t) outdiv)<<17) \
641 + (((uint32_t) 1)<<18)) >> 19);
642
643 cmd_rx_hop(
644 (fc>>19) - 1,
645 (fc & 0x7ffff) | 0x80000,
646 vco_cnt
647 );
648 wait_for_cts();
649 EX0_RESTORE;
650}

682static void
683send_bulk_conf(__code uint8_t * __pdata ids, __code uint8_t * __pdata data)
684{
685 register uint8_t len;
686 while ((len = *ids++) > 0) {
687 NSS1 = 0;
688 exchange_byte(0x11);
689 exchange_byte(*ids++); /* group id */
690 exchange_byte(len);
691 exchange_byte(*ids++); /* property id start */
692 while (len--)
693 exchange_byte(*data++);
694 NSS1 = 1;
695 wait_for_cts();
696 }
697}

699// configure radio based on the air data rate
700//
701bool
702radio_configure(__pdata uint8_t air_rate)
703{
704 __pdata uint8_t i, field_flags;
705
706 for (i = 0; i < NUM_DATA_RATES - 1; i++) {
707 if (air_data_rates[i] >= air_rate) break;
708 }
709 send_bulk_conf(shared_prop_ids, shared_prop_vals);
710 if (g_board_frequency == FREQ_433) {
711 send_bulk_conf(variable_prop_ids, band_433_prop_vals[i]);
712 } else if (g_board_frequency == FREQ_868) {
713 send_bulk_conf(variable_prop_ids, band_868_prop_vals[i]);
714 } else if (g_board_frequency == FREQ_915) {
715 send_bulk_conf(variable_prop_ids, band_915_prop_vals[i]);
716 } else {
717 return false; /* unsupported band */
718 }
719
720 settings.air_data_rate = air_data_rates[i];
721 settings.preamble_length = 16;
722
723 /* preamble is 16 nibbles, preamble rx treshold at 5<<3 bits*/
724 cmd_set_property2(GROUP_PREAMBLE, 0x00, 0x10, 5<<3); // 0x08, 5<<2);
725 wait_for_cts();
726 /* preamble starts 0101, preamble length in nibbles */
727 cmd_set_property1(GROUP_PREAMBLE, 0x04, /* 0x21); */ 0x02);
728 wait_for_cts();
729
730 cmd_set_property2(GROUP_SYNC, 0x01, 0xb4, 0x2b);
731 wait_for_cts();
732
733 /* CRC-16, seed is all zeroes */
734 cmd_set_property1(GROUP_PKT, 0x00, 0x04);
735 wait_for_cts();
736
737 /* set data whitening compatible with Si1000 */
738 cmd_set_property5(GROUP_PKT, 0x01, 0x01, 0x08, 0xff, 0xfe, 0x20);
739 wait_for_cts();
740
741 /* split fields for TX and RX, invert order of CRC bytes */
742 cmd_set_property1(GROUP_PKT, 0x06, 0x82);
743 wait_for_cts();
744
745 if (!feature_golay) {
746 /* leave length byte in FIFO, field 3 will have variable length */
747 cmd_set_property1(GROUP_PKT, 0x08, 0x08 | 0x03);
748 wait_for_cts();
749
750 /* field 2 contains packet length */
751 cmd_set_property1(GROUP_PKT, 0x09, 0x02);
752 wait_for_cts();
753
754 /* common flags shared by all fields */
755 field_flags = 0x02; /* enable whitening */
756
757 /* RX field 1 (header): two bytes long, seed PN generator, enable and seed CRC */
758 cmd_set_property4(GROUP_PKT, 0x21, 0x00, 0x02, field_flags | 0x04, 0x82);
759 wait_for_cts();
760
761 /* RX field 2 (length): one byte long, enable CRC */
762 cmd_set_property4(GROUP_PKT, 0x25, 0x00, 0x01, field_flags, 0x02);
763 wait_for_cts();
764
765 /* RX field 3 (payload): variable length, enable CRC, check CRC at end of field */
766 cmd_set_property4(GROUP_PKT, 0x29, 0x00, MAX_PACKET_LENGTH, field_flags, 0x0a);
767 wait_for_cts();
768
769 /* TX field 1 (whole packet): seed PN generator, enable and seed CRC, send CRC at end of field */
770 cmd_set_property4(GROUP_PKT, 0x0d, 0x00, 0x00, field_flags | 0x04, 0xa2);
771 wait_for_cts();
772
773 /* enable match bytes 1 and 2, point them toward the header */
774 cmd_set_property3(GROUP_MATCH, 0x00, 0x00, 0xff, 0x40);
775 wait_for_cts();
776 cmd_set_property3(GROUP_MATCH, 0x03, 0x00, 0xff, 0x01);
777 wait_for_cts();
778 /* comply with requirement of 'non-descending offsets' */
779 cmd_set_property1(GROUP_MATCH, 0x08, 0x01);
780 wait_for_cts();
781 cmd_set_property1(GROUP_MATCH, 0x0b, 0x01);
782 wait_for_cts();
783 } else {
784 /* leave length byte in FIFO, field 2 will have variable length */
785 cmd_set_property1(GROUP_PKT, 0x08, 0x08 | 0x02);
786 wait_for_cts();
787
788 /* field 1 contains packet length */
789 cmd_set_property1(GROUP_PKT, 0x09, 0x01);
790 wait_for_cts();
791
792 /* common flags shared by all fields */
793 field_flags = 0x02; /* enable whitening */
794
795 /* RX field 1 (length): one byte long, no CRC */
796 cmd_set_property4(GROUP_PKT, 0x21, 0x00, 0x01, field_flags | 0x04, 0x00);
797 wait_for_cts();
798
799 /* RX field 2 (payload): variable length, no CRC */
800 cmd_set_property4(GROUP_PKT, 0x25, 0x00, MAX_PACKET_LENGTH, field_flags, 0x00);
801 wait_for_cts();
802
803 /* TX field 1 (whole packet): seed PN generator */
804 cmd_set_property4(GROUP_PKT, 0x0d, 0x00, 0x00, field_flags | 0x04, 0x00);
805 wait_for_cts();
806 }
807
808 cmd_set_property2(GROUP_PKT, 0x0b, TX_FIFO_THRESHOLD, RX_FIFO_THRESHOLD);
809 wait_for_cts();
810
811 /* latch RSSI at sync detect, no threshold, averaging AVERAGE4 */
812 cmd_set_property1(GROUP_MODEM, 0x4c, 0x02);
813 wait_for_cts();
814
815 /* clear the FIFO to apply the mode change */
816 cmd_fifo_info(0x03);
817 wait_for_cts();
818
819 radio_gpio_init();
820
821 return true;

912static uint32_t convert_freq(uint8_t outdiv, uint32_t frequency) __reentrant
913{
914 register uint8_t x, i;
915 uint32_t freq_control;
916
917 frequency *= (outdiv / 2);
918
919 freq_control = 0;
920 for (i = 0; i < 20; i++) {
921 freq_control <<= 1;
922 if (i == 19)
923 frequency += XO_FREQ / 2;
924 x = frequency / XO_FREQ;
925 freq_control += x;
926 frequency -= x * XO_FREQ;
927 frequency <<= 1;
928 }
929
930 return freq_control;
931}

933static void
934set_frequency_registers(__pdata uint32_t base_freq, __pdata uint32_t freq_spacing)
935{
936 uint8_t band;
937
938 outdiv = 2*XO_FREQ / (base_freq / 70);
939
940 band = 0;
941 if (outdiv >= 24) {
942 outdiv = 24; band = 5;
943 } else if (outdiv >= 16) {
944 outdiv = 16; band = 4;
945 } else if (outdiv >= 12) {
946 outdiv = 12; band = 3;
947 } else if (outdiv >= 8) {
948 outdiv = 8; band = 2;
949 } else if (outdiv >= 6) {
950 outdiv = 6; band = 1;
951 } else if (outdiv >= 4) {
952 outdiv = 4; band = 0;
953 } else {
954 panic("set_frequency_registers: outdiv out of range");
955 }
956
957 /* MODEM_CLKGEN_BAND */
958 cmd_set_property1(GROUP_MODEM, 0x51, 0x08 | band);
959 wait_for_cts();
960
961 freq_control_base = convert_freq(outdiv, base_freq);
962 cmd_set_property4(
963 GROUP_FREQ_CONTROL, 0x00,
964 (freq_control_base>>19) - 1,
965 ((freq_control_base>>16) & 0x7) | 0x8,
966 (freq_control_base>>8) & 0xff,
967 freq_control_base & 0xff
968 );
969 wait_for_cts();
970
971 // for compatibility with si1000
972 freq_spacing += 10000/2;
973 freq_spacing /= 10000;
974 freq_spacing *= 10000;
975
976 freq_control_spacing = (uint16_t) convert_freq(outdiv, freq_spacing);
977 cmd_set_property2(
978 GROUP_FREQ_CONTROL, 0x04,
979 (freq_control_spacing>>8) & 0xff,
980 freq_control_spacing & 0xff
981 );
982 wait_for_cts();

1020/*
1021 Interrupt handler
1022
1023 Only used when radio is receiving. The events that trigger interrupt
1024 are selected in RX_INT0_PH_MASK and RX_INT0_MODEM_MASK.
1025
1026 My reading of the docs leads me to believe that SYNC word detection must
1027 be followed by one of:
1028
1029 - Header filter miss
1030 - CRC error
1031 - Valid packet receipt
1032
1033 All of these should be accompanied by change from RX to READY state, as
1034 we requested when issuing START_RX. In the first two cases, we go back
1035 to RX state from within the interrupt handler below. In the case of valid
1036 packet, we return to RX state when the packet contents is being picked up
1037 with a call to radio_receive_packet.
1038
1039 We set sync_detected when SYNC word is detected, and clear it when
1040 reentering RX state.
1041*/
1042
1043INTERRUPT(Receiver_ISR, INTERRUPT_INT0)
1044{
1045 uint8_t modem_pend, ph_status, ph_pend;
1046 uint8_t rx_fifo_count, rx_fifo_space;
1047
1048 /* only clear pending bits we handle here */
1049 const uint8_t m1 = RX_INT0_PH_MASK;
1050 const uint8_t m2 = RX_INT0_MODEM_MASK;
1051 cmd_get_int_status(~m1, ~m2, 0xffU);
1052 get_int_status_reply(_skip, _skip, ph_pend, ph_status, modem_pend,
1053 _skip, _skip, _skip);
1054
1055 if (modem_pend & MODEM_STATUS_SYNC_DETECT) {
1056 sync_detected = 1;
1057 }
1058
1059 if ((ph_status & PH_STATUS_RX_FIFO_ALMOST_FULL) &&
1060 (ph_pend & PH_STATUS_PACKET_RX) == 0) {
1061 if (RX_FIFO_THRESHOLD + (uint16_t) partial_packet_length > MAX_PACKET_LENGTH) {
1062 debug("partial long packet %u %u", partial_packet_length, RX_FIFO_THRESHOLD);
1063 inc_rx_error();
1064 goto rxfail;
1065 }
1066
1067 read_rx_fifo(RX_FIFO_THRESHOLD, radio_buffer + partial_packet_length);
1068 partial_packet_length += RX_FIFO_THRESHOLD;
1069 }
1070
1071 if (ph_pend & PH_STATUS_FILTER_MISS) {
1072 debug("miss");
1073 goto rxfail;
1074 }
1075
1076 if (ph_pend & PH_STATUS_CRC_ERROR) {
1077 debug("CRC error");
1078 inc_rx_error();
1079 goto rxfail;
1080 }
1081
1082 if (ph_pend & PH_STATUS_PACKET_RX) {
1083 /* get packet length */
1084 cmd_packet_info(0, 0, 0);
1085 packet_info_reply(receive_packet_length);
1086
1087 /* account for header and size byte */
1088 if (feature_golay) {
1089 receive_packet_length += 1;
1090 } else {
1091 receive_packet_length += 3;
1092 }
1093
1094 /* retrieve number of bytes in FIFO */
1095 cmd_fifo_info(0);
1096 fifo_info_reply(rx_fifo_count, rx_fifo_space);
1097
1098 if (rx_fifo_count + partial_packet_length != receive_packet_length) {
1099 debug("length mismatch fifo=%x partial=%x packet=%x",
1100 rx_fifo_count, partial_packet_length, receive_packet_length);
1101 inc_rx_error();
1102 goto rxfail;
1103 }
1104
1105 if (receive_packet_length > MAX_RAW_PACKET_LENGTH) {
1106 debug("too long %u %u %u %u", receive_packet_length, partial_packet_length, rx_fifo_count, rx_fifo_space);
1107 inc_rx_error();
1108 goto rxfail;
1109 }
1110
1111 if (partial_packet_length < receive_packet_length) {
1112 read_rx_fifo(receive_packet_length - partial_packet_length,
1113 radio_buffer + partial_packet_length);
1114 }
1115
1116 frr_a_read(last_rssi);
1117 packet_received = true;
1118 clear_rx_fifo();
1119 }
1120
1121 return;
1122
1123rxfail:
1124 _radio_receiver_on();
1125}
