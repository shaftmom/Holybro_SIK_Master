59void
60flash_erase_scratch(void)
61__critical {
62 // erase the scratch page
63#ifdef CPU_SI1030 // Part of memory no seperate scratch section
64 __pdata uint8_t bank_state = PSBANK;
65 PSBANK = ((bank_state & 0x03) | 0x30); // Select Bank 3
66#endif // CPU_SI1030
67
68 flash_load_keys(); // unlock flash for one operation
69 PSCTL = FLASH_ERASE_SCRATCH; // enable flash erase of the scratch page
70 *(uint8_t __xdata *)FLASH_SCRATCH = 0xff; // trigger the erase
71 PSCTL = FLASH_DISABLE; // disable flash write & scratch access
72
73#ifdef CPU_SI1030
74 PSBANK = bank_state;
75#endif // CPU_SI1030
76}
