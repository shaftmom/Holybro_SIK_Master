43// Serial rx/tx buffers.
44//
45// Note that the rx buffer is much larger than you might expect
46// as we need the receive buffer to be many times larger than the
47// largest possible air packet size for efficient TDM. Ideally it
48// would be about 16x larger than the largest air packet if we have
49// 8 TDM time slots
50//
51
52#ifdef CPU_SI1030
53#define RX_BUFF_MAX 1024 //2048
54#define TX_BUFF_MAX 1024
55#define ENCRYPT_BUFF_MAX 17*60 // 16 bit encrypted packets plus one for size
56static __pdata uint16_t encrypt_buff_start = 400; // Start decrypting more to clear buffer
57static __pdata uint16_t encrypt_buff_end = 500; // End our quick buffer clear
58#else
59#define RX_BUFF_MAX 1850
60#define TX_BUFF_MAX 645
61#endif // CPU_SI1030
62
63__xdata uint8_t rx_buf[RX_BUFF_MAX] = {0};
64__xdata uint8_t tx_buf[TX_BUFF_MAX] = {0};
65#ifdef INCLUDE_AES
66__xdata uint8_t encrypt_buf[ENCRYPT_BUFF_MAX] = {0};
67#endif // INCLUDE_AES
68// FIFO insert/remove pointers
69static volatile __pdata uint16_t rx_insert, rx_remove;
70static volatile __pdata uint16_t tx_insert, tx_remove;
71#ifdef CPU_SI1030
72static volatile __pdata uint16_t encrypt_insert, encrypt_remove;
73#endif
74
75// count of number of bytes we are allowed to send due to a RTS low reading
76static uint8_t rts_count;
77
78// flag indicating the transmitter is idle
79static volatile bool tx_idle;

81// FIFO status
82#define BUF_NEXT_INSERT(_b) ((_b##_insert + 1) == sizeof(_b##_buf)?0:(_b##_insert + 1))
83#define BUF_NEXT_REMOVE(_b) ((_b##_remove + 1) == sizeof(_b##_buf)?0:(_b##_remove + 1))
84#define BUF_FULL(_b) (BUF_NEXT_INSERT(_b) == (_b##_remove))
85#define BUF_NOT_FULL(_b) (BUF_NEXT_INSERT(_b) != (_b##_remove))
86#define BUF_EMPTY(_b) (_b##_insert == _b##_remove)
87#define BUF_NOT_EMPTY(_b) (_b##_insert != _b##_remove)
88#define BUF_USED(_b) ((_b##_insert >= _b##_remove)?(_b##_insert - _b##_remove):(sizeof(_b##_buf) - _b##_remove) + _b##_insert)
89#define BUF_FREE(_b) ((_b##_insert >= _b##_remove)?(sizeof(_b##_buf) + _b##_remove - _b##_insert):_b##_remove - _b##_insert)
90
91// FIFO insert/remove operations
92//
93// Note that these are nominally interrupt-safe as only one of each
94// buffer's end pointer is adjusted by either of interrupt or regular
95// mode code. This is violated if printing from interrupt context,
96// which should generally be avoided when possible.
97//
98#define BUF_INSERT(_b, _c) do { _b##_buf[_b##_insert] = (_c); \
99 _b##_insert = BUF_NEXT_INSERT(_b); } while(0)
100#define BUF_REMOVE(_b, _c) do { (_c) = _b##_buf[_b##_remove]; \
101 _b##_remove = BUF_NEXT_REMOVE(_b); } while(0)
102#define BUF_PEEK(_b) _b##_buf[_b##_remove]
103#define BUF_PEEK2(_b) _b##_buf[BUF_NEXT_REMOVE(_b)]
104#define BUF_PEEKX(_b, offset) _b##_buf[(_b##_remove+offset) % sizeof(_b##_buf)]

120void
121serial_interrupt(void) __interrupt(INTERRUPT_UART0)
122{
123 register uint8_t c;
124
125 // check for received byte first
126 if (RI0) {
127 // acknowledge interrupt and fetch the byte immediately
128 RI0 = 0;
129 c = SBUF0;
130
131 // if AT mode is active, the AT processor owns the byte
132 if (at_mode_active) {
133 // If an AT command is ready/being processed, we would ignore this byte
134 if (!at_cmd_ready) {
135 at_input(c);
136 }
137 } else {
138 // run the byte past the +++ detector
139 at_plus_detector(c);
140
141 // and queue it for general reception
142 if (BUF_NOT_FULL(rx)) {
143 BUF_INSERT(rx, c);
144 } else {
145 if (errors.serial_rx_overflow != 0xFFFF) {
146 errors.serial_rx_overflow++;
147 }
148 }
149#ifdef SERIAL_CTS
150 if (BUF_FREE(rx) < SERIAL_CTS_THRESHOLD_LOW) {
151 SERIAL_CTS = true;
152 }
153#endif
154 }
155 }
156
157 // check for anything to transmit
158 if (TI0) {
159 // acknowledge the interrupt
160 TI0 = 0;
161
162 // look for another byte we can send
163 if (BUF_NOT_EMPTY(tx)) {
164#ifdef SERIAL_RTS
165 if (feature_rtscts) {
166 if (SERIAL_RTS && !at_mode_active) {
167 if (rts_count == 0) {
168 // the other end doesn't have room in
169 // its serial buffer
170 tx_idle = true;
171 return;
172 }
173 rts_count--;
174 } else {
175 rts_count = 8;
176 }
177 }
178#endif
179 // fetch and send a byte
180 BUF_REMOVE(tx, c);
181 SBUF0 = c;
182 } else {
183 // note that the transmitter requires a kick to restart it
184 tx_idle = true;
185 }
186 }
187}

200void
201serial_init(register uint8_t speed)
202{
203 // disable UART interrupts
204 ES0 = 0;
205
206 // reset buffer state, discard all data
207 rx_insert = 0;
208 rx_remove = 0;
209 tx_insert = 0;
210 tx_remove = 0;
211#ifdef CPU_SI1030
212 encrypt_insert = 0;
213 encrypt_remove = 0;
214#endif
215 tx_idle = true;
216
217 // configure timer 1 for bit clock generation
218 TR1 = 0; // timer off
219 TMOD = (TMOD & ~0xf0) | 0x20; // 8-bit free-running auto-reload mode
220 serial_device_set_speed(speed); // device-specific clocking setup
221 TR1 = 1; // timer on
222
223 // configure the serial port
224 SCON0 = 0x10; // enable receiver, clear interrupts
225
226#ifdef SERIAL_CTS
227 // setting SERIAL_CTS low tells the other end that we have
228 // buffer space available
229 SERIAL_CTS = false;
230#endif
231
232 // re-enable UART interrupts
233 ES0 = 1;
234}

246static void
247_serial_write(register uint8_t c) __reentrant
248{
249 ES0_SAVE_DISABLE;
250
251 // if we have space to store the character
252 if (BUF_NOT_FULL(tx)) {
253
254 // queue the character
255 BUF_INSERT(tx, c);
256
257 // if the transmitter is idle, restart it
258 if (tx_idle)
259 serial_restart();
260 } else if (errors.serial_tx_overflow != 0xFFFF) {
261 errors.serial_tx_overflow++;
262 }
263
264 ES0_RESTORE;
265}

268// If on appropriate CPU and encryption configured, then attempt to decrypt it
269bool
270decryptPackets(void)
271{
272 // Encrypted packets arn't bigger than 32 bytes
273 // Limited by packet.c packet_get_next()
274 static __pdata uint8_t len_decrypted;
275 static __xdata uint8_t decrypt_buf[32];
276
277 if(BUF_NOT_EMPTY(encrypt) && aes_get_encryption_level() > 0)
278 {
279 if (encrypt_buf[encrypt_remove] == 0)
280 {
281 __critical {
282 encrypt_remove = 0;
283 }
284 }
285 if (aes_decrypt(&encrypt_buf[encrypt_remove+1], encrypt_buf[encrypt_remove], decrypt_buf, &len_decrypted) != 0) {
286 panic("error while trying to decrypt data");
287 }
288
289 // Now send decrypted output to serial buffer
290 serial_write_buf(decrypt_buf, len_decrypted);
291
292 // zero the packet as we read it.
293 len_decrypted = encrypt_buf[encrypt_remove];
294 encrypt_buf[encrypt_remove] = 0;
295
296 // printf("eb %u:%u!%u - ea ",encrypt_remove, encrypt_insert, len_decrypted);
297 __critical {
298 encrypt_remove += len_decrypted + 1;
299 if (encrypt_remove >= sizeof(encrypt_buf)) {
300 encrypt_remove = 0;
301 }
302 }
303 // printf("%u\n",encrypt_remove);
304 return true;
305 }
306 return false;
307}

309void
310serial_decrypt_buf(__xdata uint8_t * buf, __pdata uint8_t count)
311{
312 __pdata uint16_t space;
313
314 if (aes_get_encryption_level() > 0) {
315 // write to the end of the ring buffer or front if we dont have space
316 if (count > sizeof(encrypt_buf) - (encrypt_insert + 1)) {
317 encrypt_insert = 0;
318 }
319
320 // If we don't have enough space at all then exit
321 space = encrypt_buffer_write_space();
322 if (count > space) {
323 if (errors.serial_tx_overflow != 0xFFFF) {
324 errors.serial_tx_overflow++;
325 }
326 // Have to return, it is ALL or NOTHING. Can't decrypt part of a packet
327 return;
328 }
329
330
331
332 // Insert the length of the packet
333 encrypt_buf[encrypt_insert] = count;
334 //printf("ic %u \n",count, encrypt_insert);
335 memcpy(&encrypt_buf[encrypt_insert+1], buf, count);
336
337 __critical {
338 encrypt_insert += count + 1;
339 if (encrypt_insert >= sizeof(encrypt_buf)) {
340 encrypt_insert -= sizeof(encrypt_buf);
341 }
342 }
343 // Zero the next packet for the parser.
344 encrypt_buf[encrypt_insert] = 0;
345 }
346 else {
347 serial_write_buf(buf, count);
348 }
349}

352// write as many bytes as will fit into the serial transmit buffer
353// if encryption turned on, decrypt the packet.
354void
355serial_write_buf(__xdata uint8_t * buf, __pdata uint8_t count)
356{
357 __pdata uint16_t space;
358 __pdata uint8_t n1;
359
360 if (count == 0) {
361 return;
362 }
363
364 // discard any bytes that don't fit. We can't afford to
365 // wait for the buffer to drain as we could miss a frequency
366 // hopping transition
367 space = serial_write_space();
368 if (count > space) {
369 count = space;
370 if (errors.serial_tx_overflow != 0xFFFF) {
371 errors.serial_tx_overflow++;
372 }
373 }
374
375 // write to the end of the ring buffer
376 n1 = count;
377 if (n1 > sizeof(tx_buf) - tx_insert) {
378 n1 = sizeof(tx_buf) - tx_insert;
379 }
380 memcpy(&tx_buf[tx_insert], buf, n1);
381 buf += n1;
382 count -= n1;
383 __critical {
384 tx_insert += n1;
385 if (tx_insert >= sizeof(tx_buf)) {
386 tx_insert -= sizeof(tx_buf);
387 }
388 }
389
390 // add any leftover bytes to the start of the ring buffer
391 if (count != 0) {
392 memcpy(&tx_buf[0], buf, count);
393 __critical {
394 tx_insert = count;
395 }
396 }
397 __critical {
398 if (tx_idle) {
399 serial_restart();
400 }
401 }
402}

476// read count bytes from the serial buffer. This implementation
477// tries to be as efficient as possible, while disabling interrupts
478// for as short a time as possible
479bool
480serial_read_buf(__xdata uint8_t * buf, __pdata uint8_t count)
481{
482 __pdata uint16_t n1;
483 // the caller should have already checked this,
484 // but lets be sure
485 if (count > serial_read_available()) {
486 return false;
487 }
488 // see how much we can copy from the tail of the buffer
489 n1 = count;
490 if (n1 > sizeof(rx_buf) - rx_remove) {
491 n1 = sizeof(rx_buf) - rx_remove;
492 }
493 memcpy(buf, &rx_buf[rx_remove], n1);
494 count -= n1;
495 buf += n1;
496 // update the remove marker with interrupts disabled
497 __critical {
498 rx_remove += n1;
499 if (rx_remove >= sizeof(rx_buf)) {
500 rx_remove -= sizeof(rx_buf);
501 }
502 }
503 // any more bytes to do?
504 if (count > 0) {
505 memcpy(buf, &rx_buf[0], count);
506 __critical {
507 rx_remove = count;
508 }
509 }
510
511#ifdef SERIAL_CTS
512 __critical {
513 if (BUF_FREE(rx) > SERIAL_CTS_THRESHOLD_HIGH) {
514 SERIAL_CTS = false;
515 }
516 }
517#endif
518 return true;
519}

588static
589void serial_device_set_speed(register uint8_t speed)
590{
591 uint8_t i;
592 uint8_t num_rates = ARRAY_LENGTH(serial_rates);
593
594 for (i = 0; i < num_rates; i++) {
595 if (speed == serial_rates[i].rate) {
596 break;
597 }
598 }
599 if (i == num_rates) {
600 i = 6; // 57600 default
601 }
602
603 // set the rates in the UART
604 TH1 = serial_rates[i].th1;
605 CKCON = (CKCON & ~0x0b) | serial_rates[i].ckcon;
606
607 // tell the packet layer how fast the serial link is. This is
608 // needed for packet framing timeouts
609 packet_set_serial_speed(speed*125UL);
610}
