40// Flash memory map
41// It would be nice to derive all these numbers from some
42// simpler values, but life is easier if we just spell them all out.
43//
44#define FLASH_PAGE_SIZE 0x0400 // 1KiB
45#define FLASH_PAGE_SHIFT 10
46#define FLASH_APP_START 0x0400 // 1 page reserved for bootloader
47
48
49#if defined BOARD_rfd900u || defined BOARD_rfd900p || defined BOARD_rfd900pe
50// WARNING You need to select Bank 3 for these address to work. (p125)
51#define FLASH_INFO_PAGE 0xFC00 // 1 page reserved for bootloader (In Bank3)
52#define FLASH_LOCK_BYTE 0xFFFF // Bank3
53#define FLASH_SCRATCH 0xF800 // We don't have a scratch page so lets define one in code space (bank3).
54#define FLASH_BANKS 3 // 0-Home, 1-Bank1, 2-Bank2, 3-Bank3
55#define FLASH_SIGNATURE_BYTES (FLASH_SCRATCH - 2) // Location of the flash signature
56
57#else // BOARD_rfd900u (p144)
58
59#define FLASH_INFO_PAGE 0xf800 // 1 page reserved for bootloader
60#define FLASH_LOCK_BYTE 0xfbff
61#define FLASH_SIGNATURE_BYTES (FLASH_INFO_PAGE - 2) // Location of the flash signature
62#define FLASH_SCRATCH 0x00 // Define the start of the scratch page
63
64#endif // BOARD_rfd900u
65
66// Anticipated flash signature bytes
67//
68#define FLASH_SIG0 0x3d
69#define FLASH_SIG1 0xc2
70
71// Bootloader patchbay for frequency code
72//
73#define FLASH_FREQUENCY_BYTE (FLASH_LOCK_BYTE - 1)
74
75#if defined BOARD_rfd900a || defined BOARD_rfd900p || defined BOARD_rfd900pe
76
77// locked and unlocked areas to store power calibration info
78#define FLASH_CALIBRATION_AREA_SIZE (31)
79#define FLASH_CALIBRATION_CRC_HIGH (FLASH_FREQUENCY_BYTE - 1)
80#define FLASH_CALIBRATION_AREA_HIGH (FLASH_CALIBRATION_CRC_HIGH - FLASH_CALIBRATION_AREA_SIZE)
81#define FLASH_CALIBRATION_CRC (FLASH_SIGNATURE_BYTES - 1)
82#define FLASH_CALIBRATION_AREA (FLASH_CALIBRATION_CRC - FLASH_CALIBRATION_AREA_SIZE)
83
84#endif
