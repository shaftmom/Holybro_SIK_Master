123// board-specific hardware config
124#define HW_INIT \
125 do { \
126 /* GPIO config */ \
127 P0SKIP |= 0x0c; /* pins */ \
128 P0MDOUT |= 0x04; /* CTS pushpull */ \
129 P0MDOUT &= ~0x08; /* RTS open drain */ \
130 P0DRV &= ~0x04; /* CTS low drive */ \
131 P1SKIP |= 0x60; /* LEDs */ \
132 SFRPAGE = CONFIG_PAGE; \
133 P1DRV |= 0x60; /* LEDs */ \
134 SFRPAGE = LEGACY_PAGE; \
135 /* INT0 is the radio interrupt, on P0.7 */ \
136 IT01CF = (IT01CF & 0xf) | 0x7; \
137 IT0 = 0; /* INT0 level triggered */ \
138 } while(0)
