75// return a received packet
76//
77// returns true on success, false on no packet available
78//
79bool
80radio_receive_packet(uint8_t *length, __xdata uint8_t * __pdata buf)
81{
82#ifdef INCLUDE_GOLAY
83 __xdata uint8_t gout[3];
84 __data uint16_t crc1, crc2;
85 __data uint8_t errcount = 0;
86 __data uint8_t elen;
87#endif
88
89 if (!packet_received) {
90 return false;
91 }
92
93 if (receive_packet_length > MAX_PACKET_LENGTH) {
94 radio_receiver_on();
95 goto failed;
96 }
97
98#if 0
99 // useful for testing high packet loss
100 if ((timer_entropy() & 0x7) != 0) {
101 radio_receiver_on();
102 goto failed;
103 }
104#endif
105
106#ifdef INCLUDE_GOLAY
107 if (!feature_golay)
108#endif // INCLUDE_GOLAY
109 {
110 *length = receive_packet_length;
111 memcpy(buf, radio_buffer, receive_packet_length);
112
113 // simple unencoded packets
114 radio_receiver_on();
115 return true;
116 }
117
118#ifdef INCLUDE_GOLAY
119 // decode it in the callers buffer. This relies on the
120 // in-place decode properties of the golay code. Decoding in
121 // this way allows us to overlap decoding with the next receive
122 memcpy(buf, radio_buffer, receive_packet_length);
123
124 // enable the receiver for the next packet. This also
125 // enables the EX0 interrupt
126 elen = receive_packet_length;
127 radio_receiver_on();
128
129 if (elen < 12 || (elen%6) != 0) {
130 // not a valid length
131 debug("rx len invalid %u\n", (unsigned)elen);
132 goto failed;
133 }
134
135 // decode the header
136 errcount = golay_decode(6, buf, gout);
137 if (gout[0] != netid[0] ||
138 gout[1] != netid[1]) {
139 // its not for our network ID
140 debug("netid %x %x\n",
141 (unsigned)gout[0],
142 (unsigned)gout[1]);
143 goto failed;
144 }
145
146 if (6*((gout[2]+2)/3+2) != elen) {
147 debug("rx len mismatch1 %u %u\n",
148 (unsigned)gout[2],
149 (unsigned)elen);
150 goto failed;
151 }
152
153 // decode the CRC
154 errcount += golay_decode(6, &buf[6], gout);
155 crc1 = gout[0] | (((uint16_t)gout[1])<<8);
156
157 if (elen != 12) {
158 errcount += golay_decode(elen-12, &buf[12], buf);
159 }
160
161 *length = gout[2];
162
163 crc2 = crc16(*length, buf);
164
165 if (crc1 != crc2) {
166 debug("crc1=%x crc2=%x len=%u [%x %x]\n",
167 (unsigned)crc1,
168 (unsigned)crc2,
169 (unsigned)*length,
170 (unsigned)buf[0],
171 (unsigned)buf[1]);
172 goto failed;
173 }
174
175 if (errcount != 0) {
176 if ((uint16_t)(0xFFFF - errcount) > errors.corrected_errors) {
177 errors.corrected_errors += errcount;
178 } else {
179 errors.corrected_errors = 0xFFFF;
180 }
181 if (errors.corrected_packets != 0xFFFF) {
182 errors.corrected_packets++;
183 }
184 }
185
186 return true;
187#endif // INCLUDE_GOLAY
188
189failed:
190 if (errors.rx_errors != 0xFFFF) {
191 errors.rx_errors++;
192 }
193 return false;
194}

197// write to the radios transmit FIFO
198//
199static void
200radio_write_transmit_fifo(register uint8_t n, __xdata uint8_t * __pdata buffer)
201{
202 RADIO_PAGE();
203
204 NSS1 = 0;
205 SPIF1 = 0;
206 SPI1DAT = (0x80 | EZRADIOPRO_FIFO_ACCESS);
207
208 while (n--) {
209 while (!TXBMT1) /* noop */;
210 SPI1DAT = *buffer++;
211 }
212
213 while (!TXBMT1) /* noop */;
214 while ((SPI1CFG & 0x80) == 0x80);
215
216 SPIF1 = 0;
217 NSS1 = 1;
218
219 SFRPAGE = LEGACY_PAGE;
220}

304// simple transmit with no golay
305//
306// @param length number of data bytes to send
307// @param timeout_ticks number of 16usec RTC ticks to allow
308// for the send
309//
310// @return true if packet sent successfully
311//
312static bool
313radio_transmit_simple(__data uint8_t length, __xdata uint8_t * __pdata buf, __pdata uint16_t timeout_ticks)
314{
315 __pdata uint16_t tstart;
316 bool transmit_started;
317 __data uint8_t n;
318
319 if (length > sizeof(radio_buffer)) {
320 panic("oversized packet");
321 }
322
323 radio_clear_transmit_fifo();
324
325 register_write(EZRADIOPRO_TRANSMIT_PACKET_LENGTH, length);
326
327 // put packet in the FIFO
328 n = length;
329 if (n > TX_FIFO_THRESHOLD_LOW) {
330 n = TX_FIFO_THRESHOLD_LOW;
331 }
332 radio_write_transmit_fifo(n, buf);
333 length -= n;
334 buf += n;
335
336 // no interrupts
337 register_write(EZRADIOPRO_INTERRUPT_ENABLE_1, 0);
338 register_write(EZRADIOPRO_INTERRUPT_ENABLE_2, 0);
339
340 preamble_detected = 0;
341 transmit_started = false;
342
343 // start TX
344 register_write(EZRADIOPRO_OPERATING_AND_FUNCTION_CONTROL_1, EZRADIOPRO_TXON | EZRADIOPRO_XTON);
345#ifdef DEBUG_PINS_RADIO_TX_RX
346 P1 |= 0x01;
347#endif // DEBUG_PINS_RADIO_TX_RX
348
349 // wait for transmit complete or timeout
350 tstart = timer2_tick();
351 while ((uint16_t)(timer2_tick() - tstart) < timeout_ticks) {
352 __data uint8_t status;
353
354 // see if we can put some more bytes into the FIFO
355 status = register_read(EZRADIOPRO_INTERRUPT_STATUS_1);
356 if (transmit_started && length != 0 && (status & EZRADIOPRO_ITXFFAEM)) {
357 // the FIFO is below the low threshold. We
358 // should be able to put in
359 // 64-TX_FIFO_THRESHOLD_LOW more bytes, but
360 // it seems that this gives us an occasional
361 // fifo overflow error, so put in just 4 bytes
362 // at a time
363 n = 4;
364 if (n > length) {
365 n = length;
366 }
367 radio_write_transmit_fifo(n, buf);
368 length -= n;
369 buf += n;
370 continue;
371 }
372 if (transmit_started && length != 0 && (status & EZRADIOPRO_ITXFFAFULL) == 0) {
373 // the FIFO is below the high threshold. See
374 // comment above on how many bytes we add to
375 // the FIFO
376 n = 4;
377 if (n > length) {
378 n = length;
379 }
380 radio_write_transmit_fifo(n, buf);
381 length -= n;
382 buf += n;
383 continue;
384 }
385
386 if (status & EZRADIOPRO_IFFERR) {
387 // we ran out of bytes in the FIFO
388 radio_clear_transmit_fifo();
389 debug("FFERR %u\n", (unsigned)length);
390 if (errors.tx_errors != 0xFFFF) {
391 errors.tx_errors++;
392 }
393#ifdef DEBUG_PINS_RADIO_TX_RX
394 P1 &= ~0x01;
395#endif // DEBUG_PINS_RADIO_TX_RX
396 return false;
397 }
398
399 // the interrupt status bits only become valid once
400 // the transmitter is in full tx state
401 status = register_read(EZRADIOPRO_DEVICE_STATUS);
402 if (status & 0x02) {
403 // the chip power status is in TX mode
404 transmit_started = true;
405 continue;
406 }
407 if (transmit_started && (status & 0x02) == 0) {
408 // transmitter has finished. See if we got the
409 // whole packet out
410 if (length != 0) {
411 debug("TX short %u\n", (unsigned)length);
412 if (errors.tx_errors != 0xFFFF) {
413 errors.tx_errors++;
414 }
415#ifdef DEBUG_PINS_RADIO_TX_RX
416 P1 &= ~0x01;
417#endif // DEBUG_PINS_RADIO_TX_RX
418 return false;
419 }
420#ifdef DEBUG_PINS_RADIO_TX_RX
421 P1 &= ~0x01;
422#endif // DEBUG_PINS_RADIO_TX_RX
423 return true;
424 }
425
426 }
427#ifdef DEBUG_PINS_RADIO_TX_RX
428 P1 &= ~0x01;
429#endif // DEBUG_PINS_RADIO_TX_RX
430
431 // transmit timeout ... clear the FIFO
432 debug("TX timeout %u ts=%u tn=%u len=%u\n",
433 timeout_ticks,
434 tstart,
435 timer2_tick(),
436 (unsigned)length);
437 if (errors.tx_errors != 0xFFFF) {
438 errors.tx_errors++;
439 }
440
441 return false;
442}

444#ifdef INCLUDE_GOLAY
445// start transmitting a packet from the transmit FIFO
446//
447// @param length number of data bytes to send
448// @param timeout_ticks number of 16usec RTC ticks to allow
449// for the send
450//
451// @return true if packet sent successfully
452//
453static bool
454radio_transmit_golay(uint8_t length, __xdata uint8_t * __pdata buf, __pdata uint16_t timeout_ticks)
455{
456 __pdata uint16_t crc;
457 __xdata uint8_t gin[3];
458 __pdata uint8_t elen, rlen;
459
460 if (length > (sizeof(radio_buffer)/2)-6) {
461 debug("golay packet size %u\n", (unsigned)length);
462 panic("oversized golay packet");
463 }
464
465 // rounded length
466 rlen = ((length+2)/3)*3;
467
468 // encoded length
469 elen = (rlen+6)*2;
470
471 // start of packet is network ID and packet length
472 gin[0] = netid[0];
473 gin[1] = netid[1];
474 gin[2] = length;
475
476 // golay encode the header
477 golay_encode(3, gin, radio_buffer);
478
479 // next add a CRC, we round to 3 bytes for simplicity, adding
480 // another copy of the length in the spare byte
481 crc = crc16(length, buf);
482 gin[0] = crc&0xFF;
483 gin[1] = crc>>8;
484 gin[2] = length;
485
486 // golay encode the CRC
487 golay_encode(3, gin, &radio_buffer[6]);
488
489 // encode the rest of the payload
490 golay_encode(rlen, buf, &radio_buffer[12]);
491
492 return radio_transmit_simple(elen, radio_buffer, timeout_ticks);
493}
494#endif // INCLUDE_GOLAY

496// start transmitting a packet from the transmit FIFO
497//
498// @param length number of data bytes to send
499// @param timeout_ticks number of 16usec RTC ticks to allow
500// for the send
501//
502// @return true if packet sent successfully
503//
504bool
505radio_transmit(uint8_t length, __xdata uint8_t * __pdata buf, __pdata uint16_t timeout_ticks)
506{
507 bool ret;
508
509 EX0_SAVE_DISABLE;
510
511#if defined BOARD_rfd900a || defined BOARD_rfd900p
512 PA_ENABLE = 1; // Set PA_Enable to turn on PA prior to TX cycle
513#endif
514
515#ifdef INCLUDE_GOLAY
516 if (!feature_golay) {
517 ret = radio_transmit_simple(length, buf, timeout_ticks);
518 } else {
519 ret = radio_transmit_golay(length, buf, timeout_ticks);
520 }
521#else
522 ret = radio_transmit_simple(length, buf, timeout_ticks);
523#endif // INCLUDE_GOLAY
524
525#if defined BOARD_rfd900a || defined BOARD_rfd900p
526 PA_ENABLE = 0; // Set PA_Enable to off the PA after TX cycle
527#endif
528 EX0_RESTORE;
529 return ret;
530}

533// put the radio in receive mode
534//
535bool
536radio_receiver_on(void)
537{
538 EX0 = 0;
539
540 packet_received = 0;
541 receive_packet_length = 0;
542 preamble_detected = 0;
543 partial_packet_length = 0;
544
545 // enable receive interrupts
546 register_write(EZRADIOPRO_INTERRUPT_ENABLE_1, RADIO_RX_INTERRUPTS);
547 register_write(EZRADIOPRO_INTERRUPT_ENABLE_2, EZRADIOPRO_ENPREAVAL);
548
549 clear_status_registers();
550 radio_clear_transmit_fifo();
551 radio_clear_receive_fifo();
552
553 // put the radio in receive mode
554 register_write(EZRADIOPRO_OPERATING_AND_FUNCTION_CONTROL_1, EZRADIOPRO_RXON | EZRADIOPRO_XTON);
555
556 // enable receive interrupt
557 EX0 = 1;
558
559 return true;
560}

563// initialise the radio hardware
564//
565bool
566radio_initialise(void)
567{
568 uint8_t status;
569
570 delay_msec(50);
571
572 // make sure there is a radio on the SPI bus
573 status = register_read(EZRADIOPRO_DEVICE_VERSION);
574 if (status == 0xFF || status < 5) {
575 // no valid radio there?
576 return false;
577 }
578
579 // Reset the radio and setup all the registers
580 software_reset();
581
582 status = register_read(EZRADIOPRO_DEVICE_VERSION);
583
584 if ((status & EZRADIOPRO_IPOR) == 0) {
585 // it hasn't powered up cleanly, reset it
586 return software_reset();
587 }
588
589 if (status & EZRADIOPRO_ICHIPRDY) {
590 // already ready
591 return true;
592 }
593
594 return false;
595}

729// configure radio based on the air data rate
730//
731bool
732radio_configure(__pdata uint8_t air_rate)
733{
734 __pdata uint8_t i, rate_selection, control;
735
736 // disable interrupts
737 register_write(EZRADIOPRO_INTERRUPT_ENABLE_1, 0x00);
738 register_write(EZRADIOPRO_INTERRUPT_ENABLE_2, 0x00);
739
740 clear_status_registers();
741
742#ifdef ENABLE_RF_SWITCH
743 //set GPIO0 to GND
744 register_write(EZRADIOPRO_GPIO0_CONFIGURATION, 0x14); // RX data (output)
745 //set GPIO1 & GPIO2 to control the TRX switch
746 register_write(EZRADIOPRO_GPIO1_CONFIGURATION, 0x12); // TX state (output)
747 register_write(EZRADIOPRO_GPIO2_CONFIGURATION, 0x15); // RX state (output)
748#elif ENABLE_RFM50_SWITCH
749 //set GPIO0 & GPIO1 to control the TRX switch
750 register_write(EZRADIOPRO_GPIO0_CONFIGURATION, 0x15); // RX state (output)
751 register_write(EZRADIOPRO_GPIO1_CONFIGURATION, 0x12); // TX state (output)
752 //set GPIO2 to GND
753#elif ENABLE_RFD900_SWITCH
754 register_write(EZRADIOPRO_GPIO0_CONFIGURATION, 0x15); // RX data (output)
755 register_write(EZRADIOPRO_GPIO1_CONFIGURATION, 0x12); // RX data (output)
756#if RFD900_DIVERSITY
757 radio_set_diversity(DIVERSITY_ENABLED);
758#else
759 radio_set_diversity(DIVERSITY_DISABLED);
760#endif
761#else
762 //set GPIOx to GND
763 register_write(EZRADIOPRO_GPIO0_CONFIGURATION, 0x14); // RX data (output)
764 register_write(EZRADIOPRO_GPIO1_CONFIGURATION, 0x14); // RX data (output)
765 register_write(EZRADIOPRO_GPIO2_CONFIGURATION, 0x14); // RX data (output)
766#endif
767
768 // set capacitance
769 register_write(EZRADIOPRO_CRYSTAL_OSCILLATOR_LOAD_CAPACITANCE, EZRADIOPRO_OSC_CAP_VALUE);
770
771 // see Si1000.pdf section 23.3.8
772 if (air_rate > 100) {
773 register_write(EZRADIOPRO_CHARGEPUMP_CURRENT_TRIMMING_OVERRIDE, 0xC0);
774 }
775
776 // setup frequency and channel spacing
777 set_frequency_registers(settings.frequency);
778 register_write(EZRADIOPRO_FREQUENCY_HOPPING_STEP_SIZE, settings.channel_spacing);
779
780 if (feature_golay) {
781 // when using golay encoding we use our own crc16
782 // instead of the hardware CRC, as we need to correct
783 // bit errors before checking the CRC
784 register_write(EZRADIOPRO_DATA_ACCESS_CONTROL,
785 EZRADIOPRO_ENPACTX |
786 EZRADIOPRO_ENPACRX);
787 // 2 sync bytes and no header bytes
788 register_write(EZRADIOPRO_HEADER_CONTROL_2, EZRADIOPRO_HDLEN_0BYTE | EZRADIOPRO_SYNCLEN_2BYTE);
789
790 // no header check
791 register_write(EZRADIOPRO_HEADER_CONTROL_1, 0x00);
792 } else {
793 register_write(EZRADIOPRO_DATA_ACCESS_CONTROL,
794 EZRADIOPRO_ENPACTX |
795 EZRADIOPRO_ENPACRX |
796 EZRADIOPRO_ENCRC |
797 EZRADIOPRO_CRC_16);
798 // 2 sync bytes and 2 header bytes
799 register_write(EZRADIOPRO_HEADER_CONTROL_2, EZRADIOPRO_HDLEN_2BYTE | EZRADIOPRO_SYNCLEN_2BYTE);
800 // check 2 bytes of header
801 register_write(EZRADIOPRO_HEADER_CONTROL_1, 0x0C);
802 register_write(EZRADIOPRO_HEADER_ENABLE_3, 0xFF);
803 register_write(EZRADIOPRO_HEADER_ENABLE_2, 0xFF);
804 }
805
806
807 // set FIFO limits to allow for sending larger than 64 byte packets
808 register_write(EZRADIOPRO_TX_FIFO_CONTROL_1, TX_FIFO_THRESHOLD_HIGH);
809 register_write(EZRADIOPRO_TX_FIFO_CONTROL_2, TX_FIFO_THRESHOLD_LOW);
810 register_write(EZRADIOPRO_RX_FIFO_CONTROL, RX_FIFO_THRESHOLD_HIGH);
811
812 settings.preamble_length = 16;
813
814 register_write(EZRADIOPRO_PREAMBLE_LENGTH, settings.preamble_length); // nibbles
815 register_write(EZRADIOPRO_PREAMBLE_DETECTION_CONTROL, 5<<3); // 5 nibbles
816
817 // setup minimum output power during startup
818 radio_set_transmit_power(0);
819
820 // work out which register table column we will use
821 for (i = 0; i < NUM_DATA_RATES - 1; i++) {
822 if (air_data_rates[i] >= air_rate) break;
823 }
824 rate_selection = i;
825
826 settings.air_data_rate = air_data_rates[rate_selection];
827
828 if (settings.air_data_rate >= 32) {
829 control = 0x0D;
830 } else {
831 control = 0x2D;
832 }
833 if (param_get(PARAM_MANCHESTER) && settings.air_data_rate <= 128) {
834 // manchester encoding is not possible at above 128kbps
835 control |= EZRADIOPRO_ENMANCH;
836 }
837 register_write(EZRADIOPRO_MODULATION_MODE_CONTROL_1, control);
838
839 register_write(EZRADIOPRO_MODULATION_MODE_CONTROL_2, 0x23);
840
841 // note that EZRADIOPRO_AFCBD does not seem to work, which
842 // is a pity!
843 register_write(EZRADIOPRO_AFC_LOOP_GEARSHIFT_OVERRIDE, 0x44);
844
845 // this follows the recommendation in the register spreadsheet
846 // for AFC enabled and manchester disabled
847 if (settings.air_data_rate < 200) {
848 register_write(EZRADIOPRO_AFC_TIMING_CONTROL, 0x0A);
849 } else {
850 register_write(EZRADIOPRO_AFC_TIMING_CONTROL, 0x02);
851 }
852
853 // set the registers from the tables
854 if (g_board_frequency == FREQ_433) {
855 for (i = 0; i < NUM_RADIO_REGISTERS; i++) {
856 register_write(reg_index[i],
857 reg_table_433[i][rate_selection]);
858 }
859 } else if (g_board_frequency == FREQ_470) {
860 for (i = 0; i < NUM_RADIO_REGISTERS; i++) {
861 register_write(reg_index[i],
862 reg_table_470[i][rate_selection]);
863 }
864 } else if (g_board_frequency == FREQ_868) {
865 for (i = 0; i < NUM_RADIO_REGISTERS; i++) {
866 register_write(reg_index[i],
867 reg_table_868[i][rate_selection]);
868 }
869 } else {
870 for (i = 0; i < NUM_RADIO_REGISTERS; i++) {
871 register_write(reg_index[i],
872 reg_table_915[i][rate_selection]);
873 }
874 }
875
876 return true;
877}

958static void
959register_write(uint8_t reg, uint8_t value) __reentrant
960{
961 EX0_SAVE_DISABLE;
962
963 RADIO_PAGE();
964 NSS1 = 0; // drive NSS low
965 SPIF1 = 0; // clear SPIF
966 SPI1DAT = (reg | 0x80); // write reg address
967 while (!TXBMT1); // wait on TXBMT
968 SPI1DAT = value; // write value
969 while (!TXBMT1); // wait on TXBMT
970 while ((SPI1CFG & 0x80) == 0x80); // wait on SPIBSY
971
972 SPIF1 = 0; // leave SPIF cleared
973 NSS1 = 1; // drive NSS high
974 SFRPAGE = LEGACY_PAGE;
975
976 EX0_RESTORE;
977}

985static uint8_t
986register_read(uint8_t reg) __reentrant
987{
988 register uint8_t value;
989 EX0_SAVE_DISABLE;
990
991 RADIO_PAGE();
992 NSS1 = 0; // dsrive NSS low
993 SPIF1 = 0; // clear SPIF
994 SPI1DAT = (reg); // write reg address
995 while (!TXBMT1); // wait on TXBMT
996 SPI1DAT = 0x00; // write anything
997 while (!TXBMT1); // wait on TXBMT
998 while ((SPI1CFG & 0x80) == 0x80); // wait on SPIBSY
999 value = SPI1DAT; // read value
1000 SPIF1 = 0; // leave SPIF cleared
1001 NSS1 = 1; // drive NSS high
1002 SFRPAGE = LEGACY_PAGE;
1003
1004 EX0_RESTORE;
1005

1012static void
1013read_receive_fifo(register uint8_t n, __xdata uint8_t * buf) __reentrant
1014{
1015 RADIO_PAGE();
1016 NSS1 = 0; // drive NSS low
1017 SPIF1 = 0; // clear SPIF
1018 SPI1DAT = EZRADIOPRO_FIFO_ACCESS;
1019 while (!SPIF1); // wait on SPIF
1020 ACC = SPI1DAT; // discard first byte
1021
1022 while (n--) {
1023 SPIF1 = 0; // clear SPIF
1024 SPI1DAT = 0x00; // write anything
1025 while (!SPIF1); // wait on SPIF
1026 *buf++ = SPI1DAT; // copy to buffer
1027 }
1028
1029 SPIF1 = 0; // leave SPIF cleared
1030 NSS1 = 1; // drive NSS high
1031 SFRPAGE = LEGACY_PAGE;
1032}

1060software_reset(void)
1061{
1062 uint8_t status;
1063
1064 // Clear interrupt enable and interrupt flag bits
1065 register_write(EZRADIOPRO_INTERRUPT_ENABLE_1, 0);

1072

1075 while (IRQ) {
1076 if (delay_expired()) {
1077 return false;
1078 }

1083 register_write(EZRADIOPRO_INTERRUPT_ENABLE_2, EZRADIOPRO_ENCHIPRDY);

1086 while (!delay_expired()) {
1087 status = register_read(EZRADIOPRO_INTERRUPT_STATUS_1);
1088 status = register_read(EZRADIOPRO_INTERRUPT_STATUS_2);
1089 if (status & EZRADIOPRO_ICHIPRDY) {
1090 return true;
1091 }
1092 }
1093 return false;

1099static void
1100set_frequency_registers(__pdata uint32_t frequency)
1101{
1102 uint8_t band;
1103 __xdata uint16_t carrier;
1104
1105 if (frequency > 480000000UL) {
1106 frequency -= 480000000UL;
1107 band = frequency / 20000000UL;
1108 frequency -= (uint32_t)band * 20000000UL;
1109 frequency = scale_uint32(frequency, 625);
1110 frequency <<= 1;
1111 band |= EZRADIOPRO_HBSEL;
1112 } else {
1113 frequency -= 240000000UL;
1114 band = frequency / 10000000UL;
1115 frequency -= (uint32_t)band * 10000000UL;
1116 frequency = scale_uint32(frequency, 625);
1117 frequency <<= 2;
1118 }
1119
1120 band |= EZRADIOPRO_SBSEL;
1121 carrier = (uint16_t)frequency;
1122
1123 register_write(EZRADIOPRO_FREQUENCY_BAND_SELECT, band);
1124 register_write(EZRADIOPRO_NOMINAL_CARRIER_FREQUENCY_1, carrier >> 8);
1125 register_write(EZRADIOPRO_NOMINAL_CARRIER_FREQUENCY_0, carrier & 0xFF);
