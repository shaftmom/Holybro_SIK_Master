56/*
57 * Calculates the MAVLink checksum on a packet in pbuf[]
58 * and append it after the data
59 */
60static void mavlink_crc(register uint8_t crc_extra)
61{
62 register uint8_t length = pbuf[1];
63 __xdata uint16_t sum = 0xFFFF;
64 __xdata uint8_t i, stoplen;
65
66 uint8_t offset = (seen_mavlink == 1) ? 6 : 10;
67
68 stoplen = length + offset;
69
70 // MAVLink 1.0/2.0 has an extra CRC seed
71 pbuf[length+offset] = crc_extra;
72 stoplen++;
73
74 i = 1;
75 while (i<stoplen) {
76 register uint8_t tmp;
77 tmp = pbuf[i] ^ (uint8_t)(sum&0xff);
78 tmp ^= (tmp<<4);
79 sum = (sum>>8) ^ (tmp<<8) ^ (tmp<<3) ^ (tmp>>4);
80 i++;
81 }
82
83 pbuf[length+offset] = sum&0xFF;
84 pbuf[length+offset+1] = sum>>8;
85}

88/*
89We use a hand-crafted MAVLink packet based on the following
90message definition:
91
92http://mavlink.org/messages/common#RADIO_STATUS
93
94<message name="RADIO_STATUS" id="109">
95 <description>Status generated by radio</description>
96 <field type="uint8_t" name="rssi">local signal strength</field>
97 <field type="uint8_t" name="remrssi">remote signal strength</field>
98 <field type="uint8_t" name="txbuf">percentage free space in transmit buffer</field>
99 <field type="uint8_t" name="noise">background noise level</field>
100 <field type="uint8_t" name="remnoise">remote background noise level</field>
101 <field type="uint16_t" name="rxerrors">receive errors</field>
102 <field type="uint16_t" name="fixed">count of error corrected packets</field>
103</message>
104
105Note that the wire field ordering follows the MAVLink v1.0/v2.0 spec
106
107*/

110 uint16_t fixed;
111 uint8_t rssi;
112 uint8_t remrssi;
113 uint8_t txbuf;
114 uint8_t noise;
115 uint8_t remnoise;
116};

130bool MAVLink_report(void)
131{
132 struct mavlink_RADIO_v10 *m;
133 uint8_t full_packet_size;
134
135 if (seen_mavlink == 1) {
136 full_packet_size = sizeof(struct mavlink_RADIO_v10)+8;
137 } else if (seen_mavlink == 2) {
138 full_packet_size = sizeof(struct mavlink_RADIO_v10)+12;
139 } else {
140 full_packet_size = sizeof(struct mavlink_PROTOCOL_VERSION_v10)+12;
141 }
142
143 if (serial_write_space() < full_packet_size) {
144 // don't cause an overflow
145 return false;
146 }
147
148 // MAVLink 1.0 encoding
149 if (seen_mavlink == 1) {
150 m = (struct mavlink_RADIO_v10 *)&pbuf[6];
151 pbuf[0] = MAVLINK10_STX;
152 pbuf[1] = sizeof(struct mavlink_RADIO_v10);
153 pbuf[2] = seqnum++;
154 pbuf[3] = RADIO_SOURCE_SYSTEM;
155 pbuf[4] = RADIO_SOURCE_COMPONENT;
156 pbuf[5] = MAVLINK_MSG_ID_RADIO_STATUS;
157
158 // MAVLink 2.0 encoding
159 } else {
160 m = (struct mavlink_RADIO_v10 *)&pbuf[10];
161 pbuf[0] = MAVLINK20_STX;
162 pbuf[1] = sizeof(struct mavlink_RADIO_v10);
163 pbuf[2] = 0;
164 pbuf[3] = 0;
165 pbuf[4] = seqnum++;
166 pbuf[5] = RADIO_SOURCE_SYSTEM;
167 pbuf[6] = RADIO_SOURCE_COMPONENT;
168
169 // If this is a protocol version request,
170 // respond accordingly
171 if (seen_mavlink == 3) {
172 pbuf[7] = MAVLINK_MSG_ID_PROTOCOL_VERSION & 0xFF;
173 pbuf[8] = (MAVLINK_MSG_ID_PROTOCOL_VERSION >> 8) & 0xFF;
174 } else {
175 pbuf[7] = MAVLINK_MSG_ID_RADIO_STATUS & 0xFF;
176 pbuf[8] = 0;
177 }
178 // Never reaching the last 4 bit with any ID
179 pbuf[9] = 0;
180 }
181
182 // PROTOCOL_VERSION
183 if (seen_mavlink == 3) {
184 struct mavlink_PROTOCOL_VERSION_v10 *v = (struct mavlink_PROTOCOL_VERSION_v10 *)&pbuf[10];
185 v->min_version = 1;
186 v->max_version = 2;
187 memset(&v->spec_version_hash[0], 0, 8);
188 memset(&v->library_version_hash[0], 0, 8);
189 mavlink_crc(MAVLINK_PROTOCOL_VERSION_CRC_EXTRA);
190
191 // RADIO_STATUS
192 } else {
193 m->rxerrors = errors.rx_errors;
194 m->fixed = errors.corrected_packets;
195 m->txbuf = serial_read_space();
196 m->rssi = statistics.average_rssi;
197 m->remrssi = remote_statistics.average_rssi;
198 m->noise = statistics.average_noise;
199 m->remnoise = remote_statistics.average_noise;
200 mavlink_crc(MAVLINK_RADIO_STATUS_CRC_EXTRA);
201 }
202
203 serial_write_buf(pbuf, full_packet_size);
204
205 return true;
206}
