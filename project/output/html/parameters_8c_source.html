122static bool
123param_check(__pdata enum ParamID id, __data uint32_t val)
124{
125 // parameter value out of range - fail
126 if (id >= PARAM_MAX)
127 return false;
128
129 switch (id) {
130 case PARAM_FORMAT:
131 return false;
132
133 case PARAM_SERIAL_SPEED:
134 return serial_device_valid_speed(val);
135
136 case PARAM_AIR_SPEED:
137 if (val > 256)
138 return false;
139 break;
140
141 case PARAM_NETID:
142 // all values are OK
143 return true;
144
145 case PARAM_TXPOWER:
146 if (val > BOARD_MAXTXPOWER)
147 return false;
148 break;
149
150 case PARAM_ECC:
151 case PARAM_OPPRESEND:
152 // boolean 0/1 only
153 if (val > 1)
154 return false;
155 break;
156
157 case PARAM_MAVLINK:
158 if (val > 2)
159 return false;
160 break;
161
162 case PARAM_MAX_WINDOW:
163 // 131 milliseconds == 0x1FFF 16 usec ticks,
164 // which is the maximum we can handle with a 13
165 // bit trailer for window remaining
166 if (val > 131)
167 return false;
168 break;
169
170 default:
171 // no sanity check for this value
172 break;
173 }
174 return true;
175}

177bool
178param_set(__data enum ParamID param, __pdata param_t value)
179{
180 // Sanity-check the parameter value first.
181 if (!param_check(param, value))
182 return false;
183
184 // some parameters we update immediately
185 switch (param) {
186 case PARAM_TXPOWER:
187 // useful to update power level immediately when range
188 // testing in RSSI mode
189 radio_set_transmit_power(value);
190 value = radio_get_transmit_power();
191 break;
192
193 case PARAM_DUTY_CYCLE:
194 // update duty cycle immediately
195 value = constrain(value, 0, 100);
196 duty_cycle = value;
197 break;
198
199 case PARAM_LBT_RSSI:
200 // update LBT RSSI immediately
201 if (value != 0) {
202 value = constrain(value, 25, 220);
203 }
204 lbt_rssi = value;
205 break;
206
207 case PARAM_MAVLINK:
208 feature_mavlink_framing = (uint8_t) value;
209 value = feature_mavlink_framing;
210 break;
211
212 case PARAM_OPPRESEND:
213 break;
214
215 case PARAM_RTSCTS:
216 feature_rtscts = value?true:false;
217 value = feature_rtscts?1:0;
218 break;
219
220 default:
221 break;
222 }
223
224 parameter_values[param] = value;
225
226 return true;
227}

237bool read_params(__xdata uint8_t * __data input, uint16_t start, uint8_t size)
238{
239 uint16_t i;
240
241 for (i = start; i < start+size; i ++)
242 input[i-start] = flash_read_scratch(i);
243
244 // verify checksum
245 if (crc16(size, input) != ((uint16_t) flash_read_scratch(i+1)<<8 | flash_read_scratch(i)))
246 return false;
247 return true;
248}

250void write_params(__xdata uint8_t * __data input, uint16_t start, uint8_t size)
251{
252 uint16_t i, checksum;
253
254 // save parameters to the scratch page
255 for (i = start; i < start+size; i ++)
256 flash_write_scratch(i, input[i-start]);
257
258 // write checksum
259 checksum = crc16(size, input);
260 flash_write_scratch(i, checksum&0xFF);
261 flash_write_scratch(i+1, checksum>>8);
262}

264bool
265param_load(void)
266__critical {
267 __pdata uint8_t i, expected;
268
269 // Start with default values
270 param_default();
271
272 // loop reading the parameters array
273 expected = flash_read_scratch(PARAM_FLASH_START);
274 if (expected > sizeof(parameter_values) || expected < 12*sizeof(param_t))
275 return false;
276
277 // read and verify params
278 if(!read_params((__xdata uint8_t *)parameter_values, PARAM_FLASH_START+1, expected))
279 return false;
280
281 // decide whether we read a supported version of the structure
282 if (param_get(PARAM_FORMAT) != PARAM_FORMAT_CURRENT) {
283 debug("parameter format %lu expecting %lu", parameters[PARAM_FORMAT], PARAM_FORMAT_CURRENT);
284 return false;
285 }
286
287 for (i = 0; i < sizeof(parameter_values); i++) {
288 if (!param_check(i, parameter_values[i])) {
289 parameter_values[i] = parameter_info[i].default_value;
290 }
291 }
292
293 // read and verify pin params
294#if PIN_MAX > 0
295 if(!read_params((__xdata uint8_t *)pin_values, PIN_FLASH_START+1, sizeof(pin_values)))
296 return false;
297#endif
298
299 // read and verify encryption params
300#ifdef INCLUDE_AES
301 if(!read_params((__xdata uint8_t *)encryption_key, PARAM_E_FLASH_START+1, sizeof(encryption_key)))
302 return false;
303#endif // INCLUDE_AES
304 return true;
305}

307void
308param_save(void)
309__critical {
310
311 // tag parameters with the current format
312 parameter_values[PARAM_FORMAT] = PARAM_FORMAT_CURRENT;
313
314 // erase the scratch space
315 flash_erase_scratch();
316
317 // write param array length
318 flash_write_scratch(PARAM_FLASH_START, sizeof(parameter_values));
319
320 // write params
321 write_params((__xdata uint8_t *)parameter_values, PARAM_FLASH_START+1, sizeof(parameter_values));
322
323 // write pin params
324#if PIN_MAX > 0
325 flash_write_scratch(PIN_FLASH_START, sizeof(pin_values));
326 write_params((__xdata uint8_t *)pin_values, PIN_FLASH_START+1, sizeof(pin_values));
327#endif
328
329 // write encryption params
330#ifdef INCLUDE_AES
331 flash_write_scratch(PARAM_E_FLASH_START, sizeof(encryption_key));
332 write_params((__xdata uint8_t *)encryption_key, PARAM_E_FLASH_START+1, sizeof(encryption_key));
333#endif // INCLUDE_AES
334

457bool
458calibration_lock() __reentrant
459{
460 uint8_t idx;
461 uint8_t crc = 0;
462
463#ifdef CPU_SI1030
464 PSBANK = 0x33;
465#endif
466
467 // check that all entries are written
468 if (flash_read_byte(FLASH_CALIBRATION_CRC_HIGH) == 0xFF)
469 {
470 for (idx=0; idx < FLASH_CALIBRATION_AREA_SIZE; idx++)
471 {
472 uint8_t cal = flash_read_byte(FLASH_CALIBRATION_AREA_HIGH + idx);
473 crc ^= cal;
474 if (cal == 0xFF)
475 {
476 printf("dBm level %u not calibrated\n",idx);
477 return false;
478 }
479 }
480
481 // write crc
482 flash_write_byte(FLASH_CALIBRATION_CRC_HIGH, crc);
483 // lock the first and last pages
484 // can only be reverted by reflashing the bootloader
485 flash_write_byte(FLASH_LOCK_BYTE, 0xFE);
486 return true;
487 }
488 return false;

549bool
550param_set_encryption_key(__xdata unsigned char *key)
551{
552 __pdata uint8_t len, key_length;
553
554 // Use the new encryption level to help with key changes before reboot
555 // Deduce key length (bytes) from level 1 -> 16, 2 -> 24, 3 -> 32
556 key_length = AES_KEY_LENGTH(param_get(PARAM_ENCRYPTION));
557 len = strlen(key);
558 // If not enough characters (2 char per byte), then set default
559 if (len < 2 * key_length ) {
560 param_set_default_encryption_key(key_length);
561 //printf("%s\n",key);
562 printf("ERROR - Key length:%u, Required %u\n",len, 2 * key_length);
563 return true;
564 } else {
565 // We have sufficient characters for the encryption key.
566 // If too many characters, then it will just ignore extra ones
567 printf("key len %d\n",key_length);
568 convert_to_hex(key, encryption_key, key_length);
569 }
570
571 return true;
572}
