103void
104main(void)
105{
106#ifdef CPU_SI1030
107 PSBANK = 0x33;
108#endif
109
110 // Stash board info from the bootloader before we let anything touch
111 // the SFRs.
112 //
113 g_board_frequency = BOARD_FREQUENCY_REG;
114 g_board_bl_version = BOARD_BL_VERSION_REG;
115
116 // Load parameters from flash or defaults
117 // this is done before hardware_init() to get the serial speed
118 if (!param_load())
119 param_default();
120
121 // setup boolean features
122 feature_mavlink_framing = param_get(PARAM_MAVLINK);
123 feature_golay = param_get(PARAM_ECC)?true:false;
124 feature_rtscts = param_get(PARAM_RTSCTS)?true:false;
125
126 // Do hardware initialisation.
127 hardware_init();
128
129 // do radio initialisation
130 radio_init();
131
132 // turn on the receiver
133 if (!radio_receiver_on()) {
134 panic("failed to enable receiver");
135 }
136
137 // Init user pins
138#if PIN_MAX > 0
139 pins_user_init();
140#endif
141
142#ifdef INCLUDE_AES
143 // Initialise Encryption
144 if (! aes_init(param_get(PARAM_ENCRYPTION))) {
145 panic("failed to initialise aes");
146 }
147#endif // INCLUDE_AES
148
149 tdm_serial_loop();
150}

154{
155 va_list ap;
156
157// puts("\n**PANIC**");
158 va_start(ap, fmt);
159 vprintf(fmt, ap);
160// puts("");

163 ES0 = 1;
164

166
167 // generate a software reset
168 RSTSRC |= (1 << 4);
169 for (;;)
170 ;
171}

173static void
174hardware_init(void)
175{
176 __xdata uint16_t i;
177
178 // Disable the watchdog timer
179 PCA0MD &= ~0x40;
180
181 // Select the internal oscillator, prescale by 1
182#ifdef CPU_SI1030
183 OSCICN |= 0x80;
184#else
185 OSCICN = 0x8F;
186#endif
187 FLSCL = 0x40;
188 CLKSEL = 0x00;
189
190 // Configure the VDD brown out detector
191 VDM0CN = 0x80;
192 for (i = 0; i < 350; i++); // Wait 100us for initialization
193 RSTSRC = 0x06; // enable brown out and missing clock reset sources
194
195#if defined CPU_SI1030
196 P0SKIP = 0xCF;
197 P1SKIP = 0xFF;
198 P2SKIP = 0x28;
199#elif defined BOARD_rfd900a // Redefine port skips to override bootloader defs
200 P0SKIP = 0xCF; // P0 UART avail on XBAR
201 P1SKIP = 0xF8; // P1 SPI1 avail on XBAR
202 P2SKIP = 0xCF; // P2 CEX0 avail on XBAR P2.4, rest GPIO
203#endif
204
205 // Configure crossbar for UART
206 P0MDOUT = 0x10; // UART Tx push-pull
207 SFRPAGE = CONFIG_PAGE;
208 P0DRV = 0x10; // UART TX
209 SFRPAGE = LEGACY_PAGE;
210 XBR0 = 0x01; // UART enable
211
212 // SPI1
213#if defined CPU_SI1030
214 XBR1 |= 0x41; // Enable SPI1 (3 wire mode) + CEX0
215 P2MDOUT |= 0xFD; // SCK1, MOSI1, & NSS1,push-pull
216#elif defined BOARD_rfd900a // Redefine port skips to override bootloader defs
217 XBR1 |= 0x41; // enable SPI in 3-wire mode + CEX0
218 P1MDOUT |= 0xF5; // SCK1, MOSI1, MISO1 push-pull
219 P2MDOUT |= 0xFF; // SCK1, MOSI1, MISO1 push-pull
220#else
221 XBR1 |= 0x40; // enable SPI in 3-wire mode
222 P1MDOUT |= 0xF5; // SCK1, MOSI1, MISO1 push-pull
223#endif
224
225 /* ------------ Config Parameters ------------ */
226 SFRPAGE = CONFIG_PAGE;
227 P1DRV |= 0xF5; // SPI signals use high-current mode, LEDs and PAEN High current drive
228
229#ifdef CPU_SI1030
230 P2DRV = 0xFD; // MOSI1, SCK1, NSS1, high-drive mode
231
232 P3MDOUT |= 0xC0; /* Leds */
233 P3DRV |= 0xC0; /* Leds */
234#else
235 P2DRV |= 0xFF;
236#endif
237
238 /* ------------ Change to radio page ------------ */
239 RADIO_PAGE();
240 SPI1CFG = 0x40; // master mode
241#if defined(BOARD_ism01a) || defined(BOARD_hb1060)
242 SPI1CN = 0x0C; // 4 wire master mode, NSS1=1
243#else
244 SPI1CN = 0x00; // 3 wire master mode
245#endif
246 SPI1CKR = 0x00; // Initialise SPI prescaler to divide-by-2 (12.25MHz, technically out of spec)
247 SPI1CN |= 0x01; // enable SPI
248 NSS1 = 1; // set NSS high
249
250 /* ------------ END of Config Parameters ------------ */
251 SFRPAGE = LEGACY_PAGE;
252
253 // Clear the radio interrupt state
254 IE0 = 0;
255
256 // initialise timers
257 timer_init();
258
259 // UART - set the configured speed
260 serial_init(param_get(PARAM_SERIAL_SPEED));
261
262 // set all interrupts to the same priority level
263 IP = 0;
264
265 // global interrupt enable
266 EA = 1;
267
268 // Turn off the 'radio running' LED and turn off the bootloader LED
269 LED_RADIO = LED_OFF;
270 LED_BOOTLOADER = LED_OFF;
271
272 // ADC system initialise for temp sensor
273 AD0EN = 1; // Enable ADC0
274 ADC0CF = 0xF9; // Set amp0gn=1 (1:1)
275 ADC0AC = 0x00;
276 ADC0MX = 0x1B; // Set ADC0MX to temp sensor
277 REF0CN = 0x07; // Define reference and enable temp sensor
278
279#if defined BOARD_rfd900a || defined BOARD_rfd900p
280 // PCA0, CEX0 setup and enable.
281 PCA0MD = 0x88;
282 PCA0PWM = 0x00;
283 PCA0CPM0 = 0x42;
284 PCA0CPH0 = 0x80;
285 PCA0CN = 0x40;
286#endif
287 XBR2 = 0x40; // Crossbar (GPIO) enable
288}

290static void
291radio_init(void)
292{
293 __xdata uint32_t freq_min, freq_max;
294 __xdata uint32_t channel_spacing;
295 __xdata uint8_t txpower;
296
297 // Do generic PHY initialisation
298 if (!radio_initialise()) {
299 panic("radio_initialise failed");
300 }
301
302 switch (g_board_frequency) {
303 case FREQ_433:
304 freq_min = 433050000UL;
305 freq_max = 434790000UL;
306 txpower = 10;
307 num_fh_channels = 10;
308 break;
309 case FREQ_470:
310 freq_min = 470000000UL;
311 freq_max = 471000000UL;
312 txpower = 10;
313 num_fh_channels = 10;
314 break;
315 case FREQ_868:
316 freq_min = 868000000UL;
317 freq_max = 870000000UL;
318 txpower = 10;
319 num_fh_channels = 10;
320 break;
321 case FREQ_915:
322 freq_min = 915000000UL;
323 freq_max = 928000000UL;
324 txpower = 20;
325 num_fh_channels = MAX_FREQ_CHANNELS;
326 break;
327 default:
328 freq_min = 0;
329 freq_max = 0;
330 txpower = 0;
331 panic("bad board frequency %d", g_board_frequency);
332 break;
333 }
334
335 if (param_get(PARAM_NUM_CHANNELS) != 0) {
336 num_fh_channels = param_get(PARAM_NUM_CHANNELS);
337 }
338 if (param_get(PARAM_MIN_FREQ) != 0) {
339 freq_min = param_get(PARAM_MIN_FREQ) * 1000UL;
340 }
341 if (param_get(PARAM_MAX_FREQ) != 0) {
342 freq_max = param_get(PARAM_MAX_FREQ) * 1000UL;
343 }
344 if (param_get(PARAM_TXPOWER) != 0) {
345 txpower = param_get(PARAM_TXPOWER);
346 }
347
348 // constrain power and channels
349 txpower = constrain(txpower, BOARD_MINTXPOWER, BOARD_MAXTXPOWER);
350 num_fh_channels = constrain(num_fh_channels, 1, MAX_FREQ_CHANNELS);
351
352 // double check ranges the board can do
353 switch (g_board_frequency) {
354 case FREQ_433:
355 freq_min = constrain(freq_min, 414000000UL, 460000000UL);
356 freq_max = constrain(freq_max, 414000000UL, 460000000UL);
357 break;
358 case FREQ_470:
359 freq_min = constrain(freq_min, 450000000UL, 490000000UL);
360 freq_max = constrain(freq_max, 450000000UL, 490000000UL);
361 break;
362 case FREQ_868:
363 freq_min = constrain(freq_min, 849000000UL, 889000000UL);
364 freq_max = constrain(freq_max, 849000000UL, 889000000UL);
365 break;
366 case FREQ_915:
367 freq_min = constrain(freq_min, 868000000UL, 935000000UL);
368 freq_max = constrain(freq_max, 868000000UL, 935000000UL);
369 break;
370 default:
371 panic("bad board frequency %d", g_board_frequency);
372 break;
373 }
374
375 if (freq_max == freq_min) {
376 freq_max = freq_min + 1000000UL;
377 }
378
379 // get the duty cycle we will use
380 duty_cycle = param_get(PARAM_DUTY_CYCLE);
381 duty_cycle = constrain(duty_cycle, 0, 100);
382 param_set(PARAM_DUTY_CYCLE, duty_cycle);
383
384 // get the LBT threshold we will use
385 lbt_rssi = param_get(PARAM_LBT_RSSI);
386 if (lbt_rssi != 0) {
387 // limit to the RSSI valid range
388 lbt_rssi = constrain(lbt_rssi, 25, 220);
389 }
390 param_set(PARAM_LBT_RSSI, lbt_rssi);
391
392 // sanity checks
393 param_set(PARAM_MIN_FREQ, freq_min/1000);
394 param_set(PARAM_MAX_FREQ, freq_max/1000);
395 param_set(PARAM_NUM_CHANNELS, num_fh_channels);
396
397 channel_spacing = (freq_max - freq_min) / (num_fh_channels+2);
398
399 // add half of the channel spacing, to ensure that we are well
400 // away from the edges of the allowed range
401 freq_min += channel_spacing/2;
402
403 // add another offset based on network ID. This means that
404 // with different network IDs we will have much lower
405 // interference
406 r_srand(param_get(PARAM_NETID));
407 if (num_fh_channels > 5) {
408 freq_min += ((unsigned long)(r_rand()*625)) % channel_spacing;
409 }
410 debug("freq low=%lu high=%lu spacing=%lu\n",
411 freq_min, freq_min+(num_fh_channels*channel_spacing),
412 channel_spacing);
413
414 // set the frequency and channel spacing
415 // change base freq based on netid
416 radio_set_frequency(freq_min, channel_spacing);
417
418 // start on a channel chosen by network ID
419 radio_set_channel(param_get(PARAM_NETID) % num_fh_channels);
420
421 // And intilise the radio with them.
422 if (!radio_configure(param_get(PARAM_AIR_SPEED)) &&
423 !radio_configure(param_get(PARAM_AIR_SPEED)) &&
424 !radio_configure(param_get(PARAM_AIR_SPEED))) {
425 panic("radio_configure failed");
426 }
427
428 // report the real air data rate in parameters
429 param_set(PARAM_AIR_SPEED, radio_air_rate());
430
431 // setup network ID
432 radio_set_network_id(param_get(PARAM_NETID));
433
434 // setup transmit power
435 radio_set_transmit_power(txpower);
436
437 // report the real transmit power in settings
438 param_set(PARAM_TXPOWER, radio_get_transmit_power());
439
440#ifdef USE_RTC
441 // initialise real time clock
442 rtc_init();
443#endif
444
445 // initialise frequency hopping system
446 fhop_init();
447
448 // initialise TDM system
449 tdm_init();
450
451 at_mode_active = false;
452}
