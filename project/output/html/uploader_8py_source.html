41 # insert the byte array into the ranges dictionary, merging as we go
42 def __insert(self, address, bytes):
43 # look for a range that immediately follows this one
44 candidate = address + len(bytes)
45 if (candidate in self.ranges):
46 # found one, remove from ranges and merge it
47 nextbytes = self.ranges.pop(candidate)
48 bytes.extend(nextbytes)
49
50 # iterate the existing ranges looking for one that precedes this
51 for candidate in self.ranges.keys():
52 prevlen = len(self.ranges[candidate])
53 if ((candidate + prevlen) == address):
54 self.ranges[candidate].extend(bytes)
55 return
