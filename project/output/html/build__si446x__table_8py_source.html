3# This script generates radio/radio_446x_conf.h
4#
5# Silabs' WDS software allows to export a header file containing register values
6# for a specific configuration of the radio. For our use we want to be able to
7# switch between couple of those configurations. That's where this script comes in.
8# It compares a set of headers generated from WDS and compiles all the register
9# values into a somewhat compressed format. As part of this format it singles out
10# registers whose value does not change between all the considered configurations.
11
12import glob
13import re

15propbytes = dict()
16skip = ['RF_MODEM_RSSI_CONTROL_1', 'RF_MODEM_RSSI_COMP_1',
17 'RF_MODEM_CLKGEN_BAND_1']
18
19for fn in glob.glob('*.h'):
20 with open(fn, 'r') as f:
21 for d in re.finditer('\n#define (RF_[^ ]+) (.+)\n', f.read()):
22 if d.group(1) in skip:
23 continue
24 bytes_ = eval('[' + d.group(2) + ']')
25 if bytes_[0] != 0x11:
26 continue
27 group, off = bytes_[1], bytes_[3]
28 # keep MODEM, MODEM_CHFLT, PA, SYNTH
29 if group < 0x20 or group >= 0x30:
30 continue
31 for i, v in enumerate(bytes_[4:]):
32 k = group*256 + (off + i)
33 if k not in propbytes:
34 propbytes[k] = dict()
35 propbytes[k][fn] = v

37def print_bytes(bytes_):

40def ids_list(indices):
41 indices = sorted(indices)
42 indices_starts = [k for k in indices if k-1 not in indices]
43 ret = []
44 for i, k in enumerate(indices_starts):
45 last = i == len(indices_starts) - 1
46 l = len([k_ for k_ in indices if k_ >= k \
47 and (last or k_ < indices_starts[i+1])])
48 while l > 0:
49 l_ = min(l, 12)
50 ret += [l_, k//256, k%256]
51 l -= l_
52 k += l_
53 return ret + [0]
