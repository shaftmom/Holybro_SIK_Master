102// board-specific hardware config
103#define HW_INIT \
104 do { \
105 SFRPAGE = CONFIG_PAGE; \
106 P3MDOUT |= 0x40; /* Led Red */ \
107 P3DRV |= 0x40; /* Led Red */ \
108 SFRPAGE = LEGACY_PAGE; \
109 /* Setup Timers */ \
110 TMOD = (TMOD & ~0xf0) | 0x20; /* TMOD: timer 1 in 8-bit autoreload */ \
111 TR1 = 1; /* START Timer1 */ \
112 TI0 = 1; /* Indicate TX0 ready */ \
113 /* INT0 is the radio interrupt, on P0.1 */ \
114 IT01CF = (IT01CF & 0xf) | 0x01;\
115 IT0 = 0; /* INT0 level triggered */ \
116 P2 = 0xFF; /* P2 bug fix for SDCC and Raisonance*/ \
117 } while(0)
118
119// application/board-specific hardware config
120#define HW_INIT_APPLICATION \
121 do { \
122 SFRPAGE = CONFIG_PAGE; \
123 P0DRV |= 0x04; /* CTS */ \
124 SFRPAGE = LEGACY_PAGE; \
125 } while(0)
